\begin{lemma} \label{lem:NestedIntervals}
    Let $\set*{a_{n}}_{n = 1}^{\infty}$ and $\set*{b_{n}}_{n = 1}^{\infty}$ be sequences in $\R$,
    and let $\set*{I_{n}}_{n = 1}^{\infty}$ be a sequence of intervals in $\R$ such that
    \begin{itemize}
        \item $I_{n} = [a_{n}, b_{n}]$ for all $n \in \N$
        \item $I_{n + 1} \subseteq I_{n} \iff a_{n} \leq a_{n + 1} \leq b_{n + 1} \leq b_{n}$ for all $n \in \N$
    \end{itemize}
    where $a_{n} \leq b_{n}$ for all $n \in \N$. Then we have
    \[
    \bigcap_{n = 1}^{\infty} I_{n} \neq \emptyset.
    \]
\end{lemma}

\begin{proof}
    Let $E$ be a set defined by
    \[
    E \coloneqq \setbuilder{a_{n}}{n \in \N}.
    \]
    Since $E$ is bounded above by $b_{1}$, there exists $\alpha = \sup E$. Then we have
    \[
    a_{m} \leq \alpha \quad \text{for all } m \in \N.
    \]
    For any $m \in \N$, we have $b_{m}$ is an upper bound of $E$ since
    \begin{itemize}
        \item $a_{n} \leq a_{m} \leq b_{m}$ for all $n \in \N$ such that $n \leq m$.
        \item $a_{n} \leq b_{n} \leq b_{m}$ for all $n \in \N$ such that $n > m$.
    \end{itemize}
    Therefore, we have
    \[
    a_{m} \leq \alpha \leq b_{m} \implies \alpha \in I_{m} \text{ for all } m \in \N.
    \]
\end{proof}

\begin{theorem} \label{thm:NestedKCells}
    Let $k \in \N$ be given. Let $\set*{a_{n, j}}_{n = 1}^{\infty}$ and $\set*{b_{n, j}}_{n = 1}^{\infty}$ be sequences
    in $\R$ for all $j = 1, 2, \dots, k$, and let $\set*{I_{n}}_{n = 1}^{\infty}$ be a sequence of $k$-cells in $\R^{k}$
    such that
    \begin{itemize}
        \item $I_{n} = \setbuilder*{\bm{x} \in \R^{k}}{a_{n, j} \leq x_{j} \leq b_{n, j}
        \text{ for all } j = 1, 2, \dots, k}$ for all $n \in \N$
        \item $I_{n + 1} \subseteq I_{n} \iff a_{n, j} \leq a_{n + 1, j} \leq b_{n + 1, j} \leq b_{n, j}$
        for all $j = 1, 2, \dots, k$ and $n \in \N$
    \end{itemize}
    where $a_{n, j} \leq b_{n, j}$ for all $j = 1, 2, \dots, k$ and $n \in \N$. Then we have
    \[
    \bigcap_{n = 1}^{\infty} I_{n} \neq \emptyset.
    \]
\end{theorem}

\begin{proof}
    Let $\set{I_{n, j}}_{n = 1}^{\infty}$ be a sequence of intervals in $\R$ defined by
    \[
    I_{n, j} = [a_{n, j}, b_{n, j}] \quad \text{for all } j = 1, 2, \dots, k \text{ and } n \in \N.
    \]
    Then we have $I_{n + 1, j} \subseteq I_{n, j}$ for all $j = 1, 2, \dots, k$
    and $n \in \N$. By \Cref{lem:NestedIntervals}, we have
    \[
    \bigcap_{n = 1}^{\infty} I_{n, j} \neq \emptyset \quad \text{for all } j = 1, 2, \dots, k.
    \]
    Let $\alpha_{j} \in \displaystyle \bigcap_{n = 1}^{\infty} I_{n, j}$ be given for all $j = 1, 2, \dots, k$. Then we have
    \[
    \alpha_{j} \in I_{n, j} \implies a_{n, j} \leq \alpha_{j} \leq b_{n, j} \quad \text{for all }
    j = 1, 2, \dots, k \text{ and } n \in \N.
    \]
    Let $\bm{\alpha} = (\alpha_{1}, \alpha_{2}, \dots, \alpha_{k}) \in \R^{k}$. Then we have
    \[
    \bm{\alpha} \in I_{n} \quad \text{for all } n \in \N \implies \bm{\alpha} \in \bigcap_{n = 1}^{\infty} I_{n}.
    \]
\end{proof}

\begin{theorem} \label{thm:KCellsAreCompact}
    Let $k \in \N$ be given. Then any $k$-cell is compact in $\paren*{\R^{k}, \norm*{\cdot}}$.
\end{theorem}

\begin{proof}
    Let $I$ be a $k$-cell in $\paren*{\R^{k}, \norm*{\cdot}}$, and let $I_{1} \coloneqq I$. Then we have
    \[
    I_{1} = \prod_{j = 1}^{k} \bracket*{a_{1, j}, b_{1, j}} \subseteq \R^{k},
    \]
    where $a_{1, j}, b_{1, j} \in \R$ with $a_{1, j} \leq b_{1, j}$ for all $j = 1, 2, \dots, k$.
    Let $\set{I_{1, j, m}}$ be an interval in $\R$ defined by
    \[
    I_{1, j, m} \coloneqq \begin{dcases}
        \bracket*{a_{1, j}, \frac{a_{1, j} + b_{1, j}}{2}} & \text{if } m = 1 \\
        \bracket*{\frac{a_{1, j} + b_{1, j}}{2}, b_{1, j}} & \text{if } m = 2
    \end{dcases} \quad \text{for all } j = 1, 2, \dots, k.
    \]
    Let $S_{k}$ be a set defined by
    \[
    S_{k} \coloneqq \set*{1, 2}^{k} = \setbuilder*{\bm{m} \in \R^{k}}{m_{j} \in \set*{1, 2}
    \text{ for all } j = 1, 2, \dots, k}.
    \]
    By \Cref{thm:ProductOfFiniteSetsIsFinite}, $S_{k}$ is finite. We have
    \[
    I_{1} = \bigcup_{\bm{m} \in S_{k}} \paren*{\prod_{j = 1}^{k} I_{1, j, m_{j}}}.
    \]
    Let $\set{G_{\alpha}}_{\alpha \in A}$ be an open cover of $I_{1}$. Assume that $I_{1}$ has no finite subcover.
    Then, for each $\bm{m} \in S_{k}$, we have
    \[
    \prod_{j = 1}^{k} I_{1, j, m_{j}} \subseteq I_{1} \subseteq \bigcup_{\alpha \in A} G_{\alpha},
    \]
    which implies that $\set{G_{\alpha}}_{\alpha \in A}$ is an open cover of $\displaystyle
    \prod_{j = 1}^{k} I_{1, j, m_{j}}$. Assume that for each $\bm{m} \in S_{k}$, we have
    \[
    \prod_{j = 1}^{k} I_{1, j, m_{j}} \text{ has a finite subcover } \set{G_{\alpha}}_{\alpha \in A_{\bm{m}}}.
    \]
    Then we have
    \[
    \prod_{j = 1}^{k} I_{1, j, m_{j}} \subseteq \bigcup_{\alpha \in A_{\bm{m}}} G_{\alpha}
    \quad \text{for all } \bm{m} \in S_{k},
    \]
    where $\displaystyle \bigcup_{\alpha \in A_{\bm{m}}} G_{\alpha}$ is open by \Cref{thm:ArbitraryUnionOfOpenSetsIsOpen}.
    Then we have
    \[
    I_{1} \subseteq \bigcup_{\bm{m} \in S_{k}} \paren*{\bigcup_{\alpha \in A_{\bm{m}}} G_{\alpha}}
    = \bigcup_{\alpha \in A} G_{\alpha},
    \]
    where $A \coloneqq \displaystyle \bigcup_{\bm{m} \in S_{k}} A_{\bm{m}}$ is finite by
    \Cref{thm:FiniteUnionOfFiniteSetsIsFinite}. Thus $I_{1}$ has a finite subcover, which is a contradiction.
    Therefore, there exists $\bm{m_{1}} \in S_{k}$ such that
    \[
    \prod_{j = 1}^{k} I_{1, j, \paren*{m_{1}}_{j}} \text{ has no finite subcover}.
    \]
    Let $I_{2}$ be a $k$-cell in $\paren*{\R^{k}, \norm*{\cdot}}$ defined by
    \[
    I_{2} \coloneqq \prod_{j = 1}^{k} \bracket{a_{2, j}, b_{2, j}} \subseteq \R^{k},
    \]
    where $a_{2, j}, b_{2, j} \in \R$ is defined by
    \begin{itemize}
        \item $a_{2, j} \coloneqq \begin{dcases}
            a_{1, j} & \text{if } \paren*{m_{1}}_{j} = 1 \\
            \frac{a_{1, j} + b_{1, j}}{2} & \text{if } \paren*{m_{1}}_{j} = 2
        \end{dcases}$
        \item $b_{2, j} \coloneqq \begin{dcases}
            \frac{a_{1, j} + b_{1, j}}{2} & \text{if } \paren*{m_{1}}_{j} = 1 \\
            b_{1, j} & \text{if } \paren*{m_{1}}_{j} = 2
        \end{dcases}$
    \end{itemize}
    for all $j = 1, 2, \dots, k$. Then, for each $j = 1, 2, \dots, k$, we have
    \[
    \bracket{a_{2, j}, b_{2, j}} = I_{1, j, \paren*{m_{1}}_{j}} \subseteq \bracket{a_{1, j}, b_{1, j}},
    \]
    which implies that
    \[
    I_{2} = \prod_{j = 1}^{k} I_{1, j, \paren*{m_{1}}_{j}} \subseteq I_{1}.
    \]
    Let $\set{I_{2, j, m}}$ be an interval in $\R$ defined by
    \[
    I_{2, j, m} \coloneqq \begin{dcases}
        \bracket*{a_{2, j}, \frac{a_{2, j} + b_{2, j}}{2}} & \text{if } m = 1 \\
        \bracket*{\frac{a_{2, j} + b_{2, j}}{2}, b_{2, j}} & \text{if } m = 2
    \end{dcases} \quad \text{for all } j = 1, 2, \dots, k.
    \]
    Then we have
    \[
    I_{2} = \bigcup_{\bm{m} \in S_{k}} \paren*{\prod_{j = 1}^{k} I_{2, j, m_{j}}}.
    \]
    We have shown that $\set{G_{\alpha}}_{\alpha \in A}$ is an open cover of $I_{2}$, and $I_{2}$ has no finite subcover.
    Then, for each $\bm{m} \in S_{k}$, we have
    \[
    \prod_{j = 1}^{k} I_{2, j, m_{j}} \subseteq I_{2} \subseteq \bigcup_{\alpha \in A} G_{\alpha},
    \]
    which implies that $\set{G_{\alpha}}_{\alpha \in A}$ is an open cover of
    $\displaystyle \prod_{j = 1}^{k} I_{2, j, m_{j}}$. Therefore, there exists $\bm{m_{2}} \in S_{k}$ such that
    \[
    \prod_{j = 1}^{k} I_{2, j, \paren*{m_{2}}_{j}} \text{ has no finite subcover}.
    \]
    By repeating this process, we can construct the following sequences:
    \begin{itemize}
        \item The sequence $\set{\bm{m_{n}}}_{n = 1}^{\infty}$ in $S_{k}$.
        \item The sequences $\set{a_{n, j}}_{n = 1}^{\infty}$ and
        $\set{b_{n, j}}_{n = 1}^{\infty}$ in $\R$ for all $j = 1, 2, \dots, k$.
        \item The sequence $\set{I_{n, j, m}}_{n = 1}^{\infty}$ of intervals in $\R$
        for all $j = 1, 2, \dots, k$ and $m = 1, 2$.
        \item The sequence $\set{I_{n}}_{n = 1}^{\infty}$ of $k$-cells in $\R^{k}$.
    \end{itemize}
    We have the following properties:
    \begin{itemize}
        \item $I_{n + 1} \subseteq I_{n}$ for all $n \in \N$.
        \item $I_{n}$ has no finite subcover for all $n \in \N$.
        \item $a_{n, j} \leq a_{n + 1, j} \leq b_{n + 1, j} \leq b_{n, j}$ for all $j = 1, 2, \dots, k$ and $n \in \N$.
        \item $b_{n + 1, j} - a_{n + 1, j} = \dfrac{1}{2}(b_{n, j} - a_{n, j}) \implies b_{n, j} - a_{n, j} =
        \dfrac{1}{2^{n - 1}}(b_{1, j} - a_{1, j})$ for all $j = 1, 2, \dots, k$ and $n \in \N$.
    \end{itemize}
    Let $\delta \coloneqq \displaystyle \sqrt{\sum_{j = 1}^{k} (b_{1, j} - a_{1, j})^{2}}$. Then we have
    \[
    \norm*{\bm{x} - \bm{y}} \leq \sqrt{\sum_{j = 1}^{k} (b_{n, j} - a_{n, j})^{2}} =
    \frac{\delta}{2^{n - 1}} \quad \text{for all } \bm{x}, \bm{y} \in I_{n} \text{ and } n \in \N.
    \]
    By \Cref{thm:NestedKCells}, we have $\displaystyle \bigcap_{n = 1}^{\infty} I_{n} \neq \emptyset$.
    Let $\bm{\beta} \in \displaystyle \bigcap_{n = 1}^{\infty} I_{n}$ be given. Then there exists $\alpha_{0} \in A$
    such that $\bm{\beta} \in G_{\alpha_{0}}$. Since $G_{\alpha_{0}}$ is open, there exists $\veps \in \R^{+}$ such that
    \[
    B_{\veps}(\bm{\beta}) \subseteq G_{\alpha_{0}}.
    \]
    By \Cref{cor:ArchimedeanPropertyCorollary}, there exists $N \in \N$
    such that $\dfrac{1}{2^{N}} < \dfrac{\veps}{2\delta}$. Since $\bm{\beta} \in I_{N}$, we have
    \[
    \norm*{\bm{x} - \bm{\beta}} \leq \frac{\delta}{2^{N - 1}} < \veps \quad \text{for all } \bm{x} \in I_{N},
    \]
    which implies that $I_{N} \subseteq B_{\veps}(\bm{\beta}) \subseteq G_{\alpha_{0}}$. Therefore, $I_{N}$ has a
    finite subcover $\set{G_{\alpha_{0}}}$, which is a contradiction. Thus $I$ has a finite subcover, and $I$ is compact.
\end{proof}

\bigskip

\begin{corollary} \label{cor:ClosedIntervalsAreCompact}
    For any $a, b \in \R$ with $a \leq b$, the closed interval $[a, b]$ is compact in $\paren*{\R, \norm*{\cdot}}$.
\end{corollary}

\begin{corollary} \label{cor:FiniteUnionOfCountableSetsIsCountable}
    Let $X_{1}, X_{2}, \dots, X_{n}$ be countable sets for some $n \in \N$. Then $\displaystyle
    \bigcup_{i = 1}^{n} X_{i}$ is countable, i.e., any finite union of countable sets is countable.
\end{corollary}

\begin{proof}
    Let $X_{n + 1}, X_{n + 2}, \dots$ be countable sets. Then, by \Cref{thm:CountableUnionOfCountableSetsIsCountable},
    $\displaystyle \bigcup_{n = 1}^{\infty} X_{n}$ is countable. Assume that $\displaystyle \bigcup_{i = 1}^{n} X_{i}$
    is finite. Then $\displaystyle X_{1} \subseteq \bigcup_{i = 1}^{n} X_{i}$ is finite by
    \Cref{thm:SubsetOfFiniteSetIsFinite}, which contradicts \Cref{thm:CountableSetsAreInfinite}. Thus
    $\displaystyle \bigcup_{i = 1}^{n} X_{i}$ is infinite. Since $\displaystyle \bigcup_{i = 1}^{n} X_{i}
    \subseteq \bigcup_{n = 1}^{\infty} X_{n}$, by \Cref{thm:InfiniteSubsetOfCountableSetIsCountable},
    $\displaystyle \bigcup_{i = 1}^{n} X_{i}$ is countable. 
\end{proof}

\begin{theorem} \label{thm:CountableUnionOfFiniteSetsIsCountable}
    Let $X$ be a countable set. If $Y_x$ is finite for each $x \in X$, then the union
    \[
    Y = \bigcup_{x \in X} Y_x
    \]
    is countable, i.e., any countable union of finite sets is countable.
\end{theorem}

\begin{proof}
    There exists a bijection $f: \N \to X$. Also, for each $x \in X$, there exists a bijection
    $g_{x}: J_{n_{x}} \to Y_x$ for some $n_{x} \in \N$. Let $h: \N \to Y$ be a function defined by
    \begin{itemize}
        \item $h(j) = g_{f(1)}(j)$ for $j = 1, 2, \cdots, n_{f(1)}$.
        \item $h\displaystyle \paren*{\sum_{i = 1}^{k - 1} n_{f(i)} + m} =
        g_{f(k)}(m)$ for each $k \in \N$ with $k \geq 2$, and $m = 1, 2, \cdots, n_{f(k)}$.
    \end{itemize}
    Then $h$ is surjective. Therefore, by \Cref{thm:SurjectionFromN}, $Y$ is countable.
\end{proof}

\begin{lemma} \label{lem:NxNIsCountable}
    $\N \times \N$ is countable.
\end{lemma}

\begin{proof}
    We will prove this lemma by explicitly constructing a bijection from $\N \times \N \to \N$.
    Let $f: \N \times \N \to \N$ be a function defined by
    \[
    f(m, n) = \frac{(m + n - 1)(m + n - 2)}{2} + m
    \]
    for each $(m, n) \in \N \times \N$. We will show that $f$ is bijective.
    \begin{itemize}
        \item \textbf{Injectivity:} Assume that $(m_{1}, n_{1}) \neq (m_{2}, n_{2})$. If $m_{1} + n_{1} = m_{2} + n_{2}$,
        then $m_{1} \neq m_{2}$, so $f(m_{1}, n_{1}) \neq f(m_{2}, n_{2})$. Otherwise, $m_{1} + n_{1} \neq m_{2} + n_{2}$.
        \\[6mm]
        Without loss of generality, assume $m_{1} + n_{1} < m_{2} + n_{2}$. Then, since
        $m_{1} + n_{1} + 1 \leq m_{2} + n_{2}$, we have
        \begin{align*}
            f(m_{1}, n_{1}) &= \dfrac{(m_{1} + n_{1} - 1)(m_{1} + n_{1} - 2)}{2} + m_{1} \\
            &\leq \dfrac{(m_{1} + n_{1} - 1)(m_{1} + n_{1} - 2)}{2} + (m_{1} + n_{1} - 1) \\
            &= \dfrac{(m_{1} + n_{1})(m_{1} + n_{1} - 1)}{2} \\
            &\leq \dfrac{(m_{2} + n_{2} - 1)(m_{2} + n_{2} - 2)}{2} \\
            &< \dfrac{(m_{2} + n_{2} - 1)(m_{2} + n_{2} - 2)}{2} + m_{2} = f(m_{2}, n_{2}),
        \end{align*}
        Thus $f(m_{1}, n_{1}) \neq f(m_{2}, n_{2})$, so $f$ is injective.
        \item \textbf{Surjectivity:} Let $k \in \N$ be given. Let $t_{k}, s_{k}$ be defined as in the proof of
        \Cref{thm:CountableUnionOfCountableSetsIsCountable}. Then we have $f(s_{k}, t_{k} - s_{k} + 1) = k$.
        Thus $f$ is surjective.
    \end{itemize}
\end{proof}

\begin{theorem} \label{thm:ProductOfTwoCountableSetsIsCountable}
    Let $X$ and $Y$ be countable sets. Then the cartesian product $X \times Y$ is countable.
\end{theorem}

\begin{proof}
    There exist bijections $f: \N \to X$ and $g: \N \to Y$. Let $h: \N \times \N \to X \times Y$ be a function defined by
    \[
    h(m, n) = (f(m), g(n)) \text{ for all } (m, n) \in \N \times \N.
    \]
    Then $h$ is bijective. By \Cref{lem:NxNIsCountable}, $X \times Y$ is countable.
\end{proof}

\begin{theorem} \label{thm:ProductOfFiniteCountableSetsIsCountable}
    Let $X_{1}, X_{2}, \dots, X_{n}$ be countable sets for some $n \in \N$. Then $\displaystyle
    \prod_{i = 1}^{n} X_{i}$ is countable, i.e., any finite product of countable sets is countable.
\end{theorem}

\begin{proof}
    Let $P(n)$ be the statement that
    \[
    X_{1}, X_{2}, \dots, X_{n} \text{ are countable sets } \implies
    \displaystyle \prod_{i = 1}^{n} X_{i} \text{ is countable}.
    \]
    \begin{itemize}
        \item \textbf{Base case:} Let $X_{1}$ be a countable set. Then $\displaystyle \prod_{i = 1}^{1} X_{i} = X_{1}$
        is countable. Thus $P(1)$ is true.
        \item \textbf{Inductive step:} Assume that $P(n)$ is true for some $n \in \N$.
        
        Let $X_{1}, X_{2}, \dots, X_{n + 1}$ be countable sets. Then we have
        \[
        \prod_{i = 1}^{n + 1} X_{i} = \paren*{\prod_{i = 1}^{n} X_{i}} \times X_{n + 1}
        \]
        By the inductive hypothesis, $\displaystyle \prod_{i = 1}^{n} X_{i}$ is countable. Then, by
        \Cref{thm:ProductOfTwoCountableSetsIsCountable}, $\displaystyle \prod_{i = 1}^{n + 1} X_{i}$ is countable.
        Thus $P(n + 1)$ is true when $P(n)$ is true.
    \end{itemize}
    By the principle of mathematical induction,
    $P(n)$ is true for all $n \in \N$.
\end{proof}

\begin{theorem} \label{thm:QIsCountable}
    $\Q$ is countable.
\end{theorem}

\begin{proof}
    Any $q \in \Q$ can be expressed as $q = \dfrac{m}{n}$ for some $m \in \Z$ and $n \in \N$.
    Let $f: \Z \times \N \to \Q$ be a function defined by
    \[
    f(m, n) = \dfrac{m}{n} \text{ for all } (m, n) \in \Z \times \N.
    \]
    Then $f$ is surjective. By \Cref{thm:ZIsCountable,thm:ProductOfTwoCountableSetsIsCountable}, $\Z \times \N$
    is countable. Then there exists a bijection $g: \N \to \Z \times \N$.
    \\[6mm]
    By \Cref{thm:PropertiesOfCompositionFunctions}, $f \circ g: \N \to \Q$ is surjective. Since $\N \subseteq \Q$,
    by \Cref{lem:NIsInfinite,thm:SubsetOfFiniteSetIsFinite}, $\Q$ is infinite. Therefore,
    by \Cref{thm:SurjectionFromN}, $\Q$ is countable.
\end{proof}

\begin{lemma} \label{lem:InfiniteProductOfFiniteSetsIsUncountable}
    $X \coloneqq \displaystyle \prod_{n = 1}^{\infty} \set*{0, 1}$ is uncountable.
\end{lemma}

\begin{proof}
    For each $n \in \N$, let $e_{n} \in X$ be a sequence defined by
    \[
    e_{nm} = \begin{dcases}
        1 & \text{if } m = n \\
        0 & \text{Otherwise}
    \end{dcases}
    \]
    Let $f: \N \to X$ be a function defined by
    \[
    f(n) = e_{n} \text{ for all } n \in \N.
    \]
    Then $f$ is injective. By \Cref{lem:NIsInfinite,thm:InjectionFromInfiniteSetImpliesInfinite}, $X$ is infinite.
    \\[6mm]
    Assume that $X$ is countable. Then, there exists a bijection $g: \N \to X$.
    Let $X = \set*{g(1), g(2), g(3), \dots}$, where
    \begin{itemize}
        \item $g(1) = (a_{11}, a_{12}, a_{13}, \dots)$.
        \item $g(2) = (a_{21}, a_{22}, a_{23}, \dots)$.
        \item[] $\vdots$
    \end{itemize}
    Let $x = (b_{1}, b_{2}, b_{3}, \dots) \in X$ be a sequence defined by
    \[
    b_{n} = 1 - a_{nn} \text{ for all } n \in \N.
    \]
    Then, $x \neq g(n)$ for all $n \in \N$, which contradicts the assumption that $g$ is surjective.
    Therefore, $X$ is uncountable.
\end{proof}

\begin{theorem} \label{thm:RIsUncountable}
    $\R$ is uncountable.
\end{theorem}

\begin{proof}
    Since $\N \subseteq \R$, by \Cref{lem:NIsInfinite,thm:SubsetOfFiniteSetIsFinite}, $\R$ is infinite.
    Let $f: \N \to [0, 1)$ be a function defined by
    \[
    f(n) = \dfrac{1}{n + 1} \text{ for all } n \in \N.
    \]
    Then $f$ is injective. By \Cref{lem:NIsInfinite,thm:InjectionFromInfiniteSetImpliesInfinite},
    $[0, 1)$ is infinite. We will prove that $[0, 1) \subseteq \R$ is uncountable,
    which implies that $\R$ is uncountable by \Cref{thm:InfiniteSubsetOfCountableSetIsCountable}.
    \\[6mm]
    Let $g: [0, 1) \to \displaystyle \prod_{n = 1}^{\infty} \set*{0, 1}$ be a function defined by
    \[
    g(x) = \paren*{a_{n}^{x}}_{n = 1}^{\infty} \text{ for all } x \in [0, 1)
    \]
    where $\paren*{a_{n}^{x}}_{n = 1}^{\infty}$ is the \emph{greedy base $2$ expansion} of $x$.
    Let $Y$ be a set defined by
    \[
    Y \coloneqq g([0, 1)) \subseteq \displaystyle \prod_{n = 1}^{\infty} \set*{0, 1}.
    \]
    Then, by \Cref{thm:SameGreedyBaseExpansionImpliesEquality,thm:PropertiesOfRestrictionFunctions},
    $g|^{Y}$ is bijective.
    \\[6mm]
    Assume that $[0, 1)$ is countable. Then there exists a bijection $h: \N \to [0, 1)$. By
    \Cref{thm:PropertiesOfCompositionFunctions}, $g|^{Y} \circ h: \N \to Y$ is bijective, so $Y$ is countable.
    Let $Z \coloneqq \displaystyle \prod_{n = 1}^{\infty} \set*{0, 1} \setminus Y$. By
    \Cref{thm:NoRealNumberMatchingGreedyBaseExpansion,thm:NoRealNumberMatchingGreedyBaseExpansionConverse}, we have
    \[
    Z = \setbuilder*{\paren*{a_{n}}_{n = 1}^{\infty} \in \displaystyle \prod_{n = 1}^{\infty} \set*{0, 1}}{a_{n} = 1
    \text{ for all } n \geq N \text{ for some } N \in \N}.
    \]
    Let $Z_{N}$ be a set defined by
    \[
    Z_{N} \coloneqq \setbuilder*{\paren*{a_{n}}_{n = 1}^{\infty} \in \displaystyle \prod_{n = 1}^{\infty}
    \set*{0, 1}}{a_{n} = 1 \text{ for all } n \geq N} \text{ for each } N \in \N.
    \]
    Let $F_{N}: \displaystyle \prod_{n = 1}^{N - 1} \set*{0, 1} \to Z_{N}$ be a function defined by
    \[
    F_{N}\paren*{\paren*{a_{n}}_{n = 1}^{N - 1}} = \paren*{a_{n}}_{n = 1}^{N - 1} \cup \paren*{1}_{n = N}^{\infty}
    \text{ for all } \paren*{a_{n}}_{n = 1}^{N - 1} \in \displaystyle \prod_{n = 1}^{N - 1} \set*{0, 1}.
    \]
    Then $F_{N}$ is surjective. By \Cref{thm:ProductOfFiniteSetsIsFinite,thm:SurjectionFromFiniteSetImpliesFinite},
    $Z_{N}$ is finite. Then, by \Cref{thm:CountableUnionOfFiniteSetsIsCountable},
    $Z = \displaystyle \bigcup_{N = 1}^{\infty} Z_{N}$ is countable.
    \\[6mm]
    By \Cref{cor:FiniteUnionOfCountableSetsIsCountable}, $\displaystyle \prod_{n = 1}^{\infty} \set*{0, 1} = Y \cup Z$
    is countable. \Cref{lem:InfiniteProductOfFiniteSetsIsUncountable} shows that this is a contradiction.
    Therefore, $[0, 1)$ is uncountable.
\end{proof}

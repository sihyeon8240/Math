                Then $g_{n+1}$ is bijective. Let $K_{n + 1}$ be a set defined by
                \[
                K_{n + 1} \coloneqq (g_{n + 1} \circ f_{n+1})^{-1}(\set*{n + 1}) \subseteq J_{m}.
                \]
                Let $f_{n}: J_{m} \to J_{n}$ be a function defined by
                \[
                f_{n}(y) = \begin{cases}
                    (g_{n+1} \circ f_{n+1})(y), & \text{if } y \in J_{m} \setminus K_{n + 1} \\
                    n, & \text{if } y \in K_{n + 1}
                \end{cases}
                \]
                Then $f_{n}$ is surjective by \Cref{thm:PropertiesOfCompositionFunctions}.
                By the inductive hypothesis, we have $m - 1 \geq n$, which implies that $m \geq n + 1$.
            \end{itemize}
            Thus $Q(n + 1)$ is true whenever $Q(n)$ is true.
        \end{itemize}
        By the principle of mathematical induction,
        $Q(n)$ is true for all $n \in \N$.
    \end{enumerate}
\end{proof}

\begin{lemma} \label{lem:PropertiesOfJmJnConverse}
    Let $J_{m}, J_{n}$ be given for some $m, n \in \N$. Then we have the following properties:
    \begin{enumerate}[label=\textbf{(\Alph*)}]
        \item If $m \leq n$, then there exists an injection $f: J_{m} \to J_{n}$.
        \item If $m \geq n$, then there exists a surjection $f: J_{m} \to J_{n}$.
        \item If $m = n$, then there exists a bijection $f: J_{m} \to J_{n}$.
    \end{enumerate}
    This lemma is the converse of \Cref{lem:PropertiesOfJmJn}.
\end{lemma}

\begin{proof}
    We will prove this lemma by explicitly constructing the functions.
    \begin{enumerate}[label=\textbf{(\Alph*)}]
        \item If $m \leq n$, let $f: J_{m} \to J_{n}$ be a function defined by
        \[
        f(i) = i \quad \text{for all } i \in J_{m}.
        \]
        Then $f$ is an injection.
        \item If $m \geq n$, let $f: J_{m} \to J_{n}$ be a function defined by
        \[
        f(i) = \begin{dcases}
            i, & \text{if } i \leq n \\
            n, & \text{if } i > n
        \end{dcases}
        \]
        Then $f$ is a surjection.
        \item If $m = n$, let $f: J_{m} \to J_{n}$ be a function defined by
        \[
        f(i) = i \quad \text{for all } i \in J_{m}.
        \]
        Then $f$ is a bijection.
    \end{enumerate}
\end{proof}

\begin{theorem} \label{thm:PropertiesOfCardinality}
    Let $X, Y$ be finite sets. Then we have the following properties:
    \begin{enumerate}[label=\textbf{(\Alph*)}]
        \item If there exists an injection $f: X \to Y$, then we have $\abs{X} \leq \abs{Y}$.
        \item If there exists a surjection $f: X \to Y$, then we have $\abs{X} \geq \abs{Y}$.
        \item If there exists a bijection $f: X \to Y$, then we have $\abs{X} = \abs{Y}$.
    \end{enumerate}
    Moreover, the converses of \textbf{(A)}, \textbf{(B)}, and \textbf{(C)} are also true.
    \\[6mm]
    The contraposition of \textbf{(A)} is called the \textbf{pigeonhole principle}, which states that
    if $\abs{X} > \abs{Y}$, then there exists no injection $f: X \to Y$.
\end{theorem}

\begin{proof}
    Let $\abs{X} = m$ and $\abs{Y} = n$ for some $m, n \in \N$. Then there exist bijections
    $f_{X}: J_{m} \to X$ and $f_{Y}: J_{n} \to Y$.
    By \Cref{thm:InverseFunctionIsBijection,thm:PropertiesOfCompositionFunctions}, the composition
    \[
    F \equiv (f_{Y})^{-1} \circ f \circ f_{X}: J_{m} \to J_{n}
    \]
    is injective, surjective, or bijective if $f$ is injective, surjective, or bijective, respectively.
    Then, by \Cref{lem:PropertiesOfJmJn}, we have the desired results:
    \begin{enumerate}[label=\textbf{(\Alph*)}]
        \item If $f$ is injective, then $F$ is injective, which implies that $m \leq n$.
        \item If $f$ is surjective, then $F$ is surjective, which implies that $m \geq n$.
        \item If $f$ is bijective, then $F$ is bijective, which implies that $m = n$.
    \end{enumerate}
    Also, by \Cref{thm:InverseFunctionIsBijection,thm:PropertiesOfCompositionFunctions}, the composition
    \[
    f \equiv f_{Y} \circ G \circ (f_{X})^{-1}: X \to Y
    \]
    is injective, surjective, or bijective if $G: J_{m} \to J_{n}$ is injective, surjective, or bijective, respectively.
    Then, by \Cref{lem:PropertiesOfJmJnConverse}, we have the desired converses:
    \begin{enumerate}[label=\textbf{(\Alph*)}]
        \item If $m \leq n$, then there exists an injective function $G: J_{m} \to J_{n}$,
        which implies that there exists an injective function $f: X \to Y$.
        \item If $m \geq n$, then there exists a surjective function $G: J_{m} \to J_{n}$,
        which implies that there exists a surjective function $f: X \to Y$.
        \item If $m = n$, then there exists a bijective function $G: J_{m} \to J_{n}$,
        which implies that there exists a bijective function $f: X \to Y$.
    \end{enumerate}
\end{proof}

\begin{lemma} \label{lem:UnionOfTwoDisjointFiniteSetsIsFinite}
    Let $X, Y$ be finite sets with $X \cap Y = \emptyset$. Then the union $X \cup Y$ is finite.
\end{lemma}

\begin{proof}
    Let $m = \abs{X}$ and $n = \abs{Y}$. There exist bijections $f: J_{m} \to X$ and $g: J_{n} \to Y$.
    Let $h: J_{m + n} \to X \cup Y$ be a function defined by
    \[
    h(i) = \begin{cases}
        f(i), & \text{if } i \leq m \\
        g(i - m), & \text{if } m < i \leq m + n
    \end{cases}
    \]
    Then $h$ is bijective, which implies that $X \cup Y$ is finite.
\end{proof}

\begin{theorem} \label{thm:UnionAndIntersectionOfTwoFiniteSetsAreFinite}
    Let $X, Y$ be finite sets. Then the union $X \cup Y$ and intersection $X \cap Y$ are finite.
\end{theorem}

\begin{proof}
    Let $A, B, C$ be sets defined by
    \begin{itemize}
        \item $A = X \setminus Y \subseteq X$.
        \item $B = Y \setminus X \subseteq Y$.
        \item $C = X \cap Y \subseteq X$.
    \end{itemize}
    Then $A, B, C$ are finite by \Cref{thm:SubsetOfFiniteSetIsFinite} and are disjoint each other. We have the followings:
    \begin{itemize}
        \item $X = (X \setminus Y) \cup (X \cap Y) = A \cup C$.
        \item $Y = (Y \setminus X) \cup (X \cap Y) = B \cup C$.
        \item $X \cup Y = ((X \setminus Y) \cup (Y \setminus X)) \cup (X \cap Y) = (A \cup B) \cup C$.
    \end{itemize}
    By \Cref{lem:UnionOfTwoDisjointFiniteSetsIsFinite}, $X, Y, X \cup Y$ are finite.
\end{proof}

\begin{theorem} \label{thm:FiniteUnionOfFiniteSetsAreFinite}
    Let $X_{1}, X_{2}, \dots, X_{n}$ be finite sets for some $n \in \N$. Then the union
    $\displaystyle \bigcup_{k = 1}^{n} X_{k}$ is finite, i.e., any finite union of finite sets is finite.
\end{theorem}

\begin{proof}
    Let $P(n)$ be the statement that
    \[
    X_{1}, X_{2}, \dots, X_{n} \text{ are finite sets} \implies \bigcup_{k = 1}^{n} X_{k} \text{ is finite}.
    \]
    \begin{itemize}
        \item \textbf{Base case:} Let $X_{1}$ be a finite set. Then
        $\displaystyle \bigcup_{k = 1}^{1} X_{k}= X_{1}$ is finite. Thus $P(1)$ is true.
        \item \textbf{Inductive step:} Assume that $P(n)$ is true for some $n \in \N$.
        
        Let $X_{1}, X_{2}, \dots, X_{n + 1}$ be finite sets. Then we have
        \[
        \bigcup_{k = 1}^{n + 1} X_{k} = \paren*{\bigcup_{k = 1}^{n} X_{k}} \cup X_{n + 1}.
        \]
        By the inductive hypothesis, $\displaystyle \bigcup_{k = 1}^{n} X_{k}$ is finite. Then, by
        \Cref{thm:UnionAndIntersectionOfTwoFiniteSetsAreFinite}, $\displaystyle \bigcup_{k = 1}^{n + 1} X_{k}$ is finite.
        Thus $P(n + 1)$ is true whenever $P(n)$ is true.
    \end{itemize}
    By the principle of mathematical induction, $P(n)$ is true for all $n \in \N$.
\end{proof}

\begin{lemma} \label{lem:ProductOfTwoFiniteSetsIsFinite}
    Let $X, Y$ be finite sets. Then the cartesian product $X \times Y$ is finite.
\end{lemma}

\begin{proof}
    Let $m = \abs{X}$ and $n = \abs{Y}$. There exist bijections $f: J_{m} \to X$ and $g: J_{n} \to Y$.
    Let $h: J_{mn} \to X \times Y$ be a function defined by
    \[
    h(i) = \paren*{f\paren*{\ceil*{\dfrac{i}{n}}}, g\paren*{i - n\paren*{\ceil*{\dfrac{i}{n}} - 1}}}
    \text{ for all } i \in J_{mn}.
    \]
    Then $h$ is bijective, which implies that $X \times Y$ is finite.
\end{proof}

\begin{theorem} \label{thm:ProductOfFiniteSetsIsFinite}
    Let $X_{1}, X_{2}, \dots, X_{n}$ be finite sets for some $n \in \N$. Then the cartesian product
    $\displaystyle \prod_{k = 1}^{n} X_{k}$ is finite.
\end{theorem}

\begin{proof}
    Let $P(n)$ be the statement that
    \[
    X_{1}, X_{2}, \dots, X_{n} \text{ are finite sets} \implies \prod_{k = 1}^{n} X_{k} \text{ is finite}.
    \]
    \begin{itemize}
        \item \textbf{Base case:} Let $X_{1}$ be a finite set. Then,
        $\displaystyle \prod_{k = 1}^{1} X_{k}= X_{1}$ is finite. Thus $P(1)$ is true.
        \item \textbf{Inductive step:} Assume that $P(n)$ is true for some $n \in \N$.
        
        Let $X_{1}, X_{2}, \dots, X_{n + 1}$ be finite sets. Then we have
        \[
        \prod_{k = 1}^{n + 1} X_{k} = \paren*{\prod_{k = 1}^{n} X_{k}} \times X_{n + 1}.
        \]
        By the inductive hypothesis, $\displaystyle \prod_{k = 1}^{n} X_{k}$ is finite. Then, by
        \Cref{lem:ProductOfTwoFiniteSetsIsFinite}, $\displaystyle \prod_{k = 1}^{n + 1} X_{k}$ is finite.
        Thus $P(n + 1)$ is true whenever $P(n)$ is true.
    \end{itemize}
    By the principle of mathematical induction, $P(n)$ is true for all $n \in \N$.
\end{proof}

\begin{theorem} \label{thm:CantorsTheorem}
    Let $X$ be a nonempty set. Then there exists no surjection $f: X \to \mathcal{P}(X)$, where
    \[
    \mathcal{P}(X) \coloneqq \setbuilder*{E}{E \subseteq X}
    \]
    is called the \textbf{power set} of $X$. This theorem is called \textbf{Cantor's theorem}.
\end{theorem}

\begin{proof}
    Let $S$ be a set defined by
    \[
    S \coloneqq \setbuilder*{x \in X}{x \notin f(x)}.
    \]
    Since $S \subseteq X$, we have that $S \in \mathcal{P}(X)$. Assume that there exists a surjection
    $f: X \to \mathcal{P}(X)$. Then there exists $x \in X$ such that $f(x) = S$.
    \\[6mm]
    If $x \in S$, then we have $x \notin f(x) = S$, which is a contradiction. If $x \notin S$, then we have
    $x \in f(x) = S$, which is also a contradiction. Therefore, there exists no surjection $f: X \to \mathcal{P}(X)$.
\end{proof}

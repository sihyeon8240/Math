\begin{section}{Connected Sets}

Let $(X, d)$ be a metric space and $E \subseteq X$. $E$ is called \textbf{connected}
if there exists no pair of sets $A, B \subseteq X$ such that
\begin{itemize}
    \item $\vnot \neq A \subseteq E$ and $\vnot \neq B \subseteq E$.
    \item $\overline{A} \cap B = A \cap \overline{B} = \vnot$.
    \item $E = A \cup B$.
\end{itemize}
Such pair of sets $A$ and $B$ is called \textbf{separated}.
If $E$ is not connected, then $E$ is called \textbf{disconnected}.

\bigskip

\begin{theorem} \label{thm:ConnectedSetsInR}
    Let $(\R, \norm*{\cdot})$ be given. Then $E \subseteq \R$ is connected if and only if
    \[
    a < x < b \implies x \in E \quad \text{for all } a, b \in E \text{ and } x \in \R.
    \]
\end{theorem}

\begin{proof}
    \textbf{($\implies$)} Let $E \subseteq \R$ be connected. Assume that there exist $a, b \in E$ and $x \in \R$ such that
    \[
    a < x < b \text{ and } x \notin E.
    \]
    Let $A, B$ be sets defined by
    \begin{itemize}
        \item $A \coloneqq E \cap (-\infty, x) \subseteq E$.
        \item $B \coloneqq E \cap (x, \infty) \subseteq E$.
    \end{itemize}
    Then we have
    \begin{itemize}
        \item $A \neq \vnot$ and $B \neq \vnot$ since $a \in A$ and $b \in B$.
        \item $\overline{A} \cap B = A \cap \overline{B} = \vnot$.
        \item $E = A \cup B$ since $x \notin E$.
    \end{itemize}
    Thus $A$ and $B$ are separated, which is a contradiction. Therefore, we have
    \[
    a < x < b \implies x \in E \quad \text{for all } a, b \in E \text{ and } x \in \R.
    \]
    \\[6mm]
    \textbf{($\impliedby$)} Let $E \subseteq \R$ be a set such that
    \[
    a < x < b \implies x \in E \quad \text{for all } a, b \in E \text{ and } x \in \R.
    \]
    Assume that $E$ is disconnected. Then there exist separated sets $A, B \subseteq \R$ such that
    \begin{itemize}
        \item $\vnot \neq A \subseteq E$ and $\vnot \neq B \subseteq E$.
        \item $\overline{A} \cap B = A \cap \overline{B} = \vnot$.
        \item $E = A \cup B$.
    \end{itemize}
    Let $a \in A$ and $b \in B$ be given. Without loss of generality, assume that $a < b$. Let
    \[
    \alpha \coloneqq \sup\paren*{A \cap [a, b]} \in \R,
    \]
    where we have
    \begin{itemize}
        \item $a \in A \implies A \cap [a, b] \neq \vnot$.
        \item $A \cap [a, b]$ is bounded above by $b$.
    \end{itemize}
    Then the \emph{existence} of $\alpha$ is guaranteed by the \emph{least upper bound property}.
    By \Cref{thm:SupremumInClosure}, we have
    \[
    \alpha \in \overline{\paren*{A \cap [a, b]}} = \paren*{A \cap [a, b]} \cup \paren*{A \cap [a, b]}'.
    \]
    By \Cref{def:LimitIsolatedInteriorPoints}, it is clear that
    \[
    \paren*{A \cap [a, b]}' \subseteq A'.
    \]
    Thus we have
    \[
    \alpha \in \paren*{A \cap [a, b]} \subseteq A \quad \text{or} \quad
    \alpha \in \paren*{A \cap [a, b]}' \subseteq A' \implies \alpha \in \overline{A}.
    \]
    Then the following statements hold:
    \begin{itemize}
        \item $\alpha = \sup\paren*{A \cap [a, b]} \implies a \leq \alpha \leq b$.
        \item $\alpha \in \overline{A} \implies \alpha \notin B \implies \alpha \neq b$.
    \end{itemize}
    If $\alpha \notin A$, then we have
    \begin{itemize}
        \item $\alpha \neq a \implies a < \alpha < b$
        \item $\alpha \notin A$ and $\alpha \notin B \implies \alpha \notin E$
    \end{itemize}
    which is a contradiction. Otherwise, we have
    \[
    \alpha \in A \implies \alpha \notin \overline{B}.
    \]
    Then there exists $r \in \paren*{0, b - \alpha}$ such that
    \[
    \widetilde{N}_{r}(\alpha) \cap B = \vnot \implies \beta \coloneqq \alpha + r \notin B.
    \]
    Then we have
    \begin{itemize}
        \item $a \leq \alpha < \beta < b$
        \item $\beta > \alpha \implies \beta \notin A \implies \beta \notin E$
    \end{itemize}
    which is a contradiction. Therefore, we conclude that $E$ is connected.
\end{proof}

\end{section}

\begin{section}{Compact Sets}

Let $(X, d)$ be a metric space and $E \subseteq X$. Let $A$ be a set and let $G_{\alpha}$
be an open set in $X$ for each $\alpha \in A$. Then the collection
$\set*{G_{\alpha}}_{\alpha \in A}$ is called an \textbf{open cover} of $E$ if
\[
E \subseteq \bigcup_{\alpha \in A} G_{\alpha}.
\]
A collection $\set*{G_{\alpha}}_{\alpha \in S}$ is called a \textbf{subcover} of $E$ if
\begin{itemize}
    \item $S \subseteq A$.
    \item $E \subseteq \displaystyle \bigcup_{\alpha \in S} G_{\alpha}$.
\end{itemize}
$E$ is called \textbf{compact} if any open cover of $E$ has a finite subcover. That is, for each
open cover $\set*{G_{\alpha}}_{\alpha \in A}$ of $E$, there exists a subset $S_{A} \subseteq A$ such that
\begin{itemize}
    \item $S_{A}$ is finite.
    \item $E \subseteq \displaystyle \bigcup_{\alpha \in S_{A}} G_{\alpha}$.
\end{itemize}

\bigskip

\begin{theorem} \label{thm:CompactnessPreservedUnderSubspace}
    Let $(X, d)$ be a metric space and $E \subseteq Y \subseteq X$.
    Then $E$ is compact relative to $Y$ if and only if $E$ is compact relative to $X$.
\end{theorem}

\begin{proof}
    \textbf{($\implies$)} Let $E$ be a compact set relative to $Y$. Let
    $\set*{G_{\alpha}}_{\alpha \in A}$ be an open cover of $E$ relative to $X$. Then we have
    \[
    E \subseteq \bigcup_{\alpha \in A} G_{\alpha}.
    \]
    Let $H_{\alpha} = G_{\alpha} \cap Y$ for each $\alpha \in A$. By \Cref{thm:OpenSetRelativeToSubspace},
    $H_{\alpha}$ is open relative to $Y$. Since $E \subseteq Y$, we have
    \[
    E \subseteq \paren*{\bigcup_{\alpha \in A} G_{\alpha}} \cap Y =
    \bigcup_{\alpha \in A} \paren*{G_{\alpha} \cap Y} = \bigcup_{\alpha \in A} H_{\alpha}.
    \]
    Thus $\set*{H_{\alpha}}_{\alpha \in A}$ is an open cover of $E$ relative to $Y$. Since $E$ is compact relative
    to $Y$, there exists a finite subcover $\set*{H_{\alpha}}_{\alpha \in S_{A}}$ of $E$ relative to $Y$. Then we have
    \[
    E \subseteq \bigcup_{\alpha \in S_{A}} H_{\alpha} = \bigcup_{\alpha \in S_{A}}
    \paren*{G_{\alpha} \cap Y} \subseteq \bigcup_{\alpha \in S_{A}} G_{\alpha}.
    \]
    Thus $\set*{G_{\alpha}}_{\alpha \in S_{A}}$ is a finite subcover of $E$ relative to $X$.
    Therefore, $E$ is compact relative to $X$.
    \\[6mm]
    \textbf{($\impliedby$)} Let $E$ be a compact set relative to $X$. Let $\set*{H_{\alpha}}_{\alpha \in A}$ be an open
    cover of $E$ relative to $Y$. Then we have
    \[
    E \subseteq \bigcup_{\alpha \in A} H_{\alpha}.
    \]
    By \Cref{thm:OpenSetRelativeToSubspace}, there exists an open set $G_{\alpha}$ in $X$ such that $H_{\alpha} =
    G_{\alpha} \cap Y$ for each $\alpha \in A$. Then we have
    \[
    E \subseteq \bigcup_{\alpha \in A} H_{\alpha} = \bigcup_{\alpha \in A} \paren*{G_{\alpha} \cap Y}
    \subseteq \bigcup_{\alpha \in A} G_{\alpha}.
    \]
    Thus $\set*{G_{\alpha}}_{\alpha \in A}$ is an open cover of $E$ relative to $X$. Since $E$ is compact relative to $X$,
    there exists a finite subcover $\set*{G_{\alpha}}_{\alpha \in S_{A}}$ of $E$ relative to $X$. Then we have
    \[
    E \subseteq \paren*{\bigcup_{\alpha \in S_{A}} G_{\alpha}} \cap Y = \bigcup_{\alpha \in S_{A}}
    \paren*{G_{\alpha} \cap Y} = \bigcup_{\alpha \in S_{A}} H_{\alpha}.
    \]
    Thus $\set*{H_{\alpha}}_{\alpha \in S_{A}}$ is a finite subcover of $E$ relative to $Y$.
    Therefore, $E$ is compact relative to $Y$.
\end{proof}

\bigskip

\begin{remark}
    The concept of \textbf{compactness} is preserved under the subspace topology, while
    the concepts of \textbf{openness} and \textbf{closedness} are not, which is the reason why
    \[
    \text{``Let $(X, d)$ be a metric space and $E \subseteq X$."}
    \]
    this kind of statement is always included in theorems and definitions about metric spaces.
    In this textbook, you can assume that
    \begin{itemize}
        \item $E$ is open. $\implies E$ is open relative to $X$.
        \item $E$ is closed. $\implies E$ is closed relative to $X$.
        \item $E$ is compact. $\implies E$ is compact relative to $X$.
    \end{itemize}
    if there is no specific mention of the subspace topology.
\end{remark}

\begin{theorem} \label{thm:CompactSetIsClosed}
    Let $(X, d)$ be a metric space and $E \subseteq X$. If $E$ is compact, then $E$ is closed.
\end{theorem}

\begin{proof}
    We will prove that $E^{c}$ is open, which implies that $E$ is closed by \Cref{cor:ClosedSetComplementIsOpenSet}.
    Let $p \in E^{c}$ be given. Since $p \notin E$, we have 
    \[
    r_{q} \coloneqq \frac{d(p, q)}{2} > 0 \quad \text{for each } q \in E.
    \]
    Since $q \in N_{r_{q}}(q)$ for each $q \in E$, we have
    \[
    E \subseteq  \bigcup_{q \in E} N_{r_{q}}(q) \implies
    \set*{N_{r_{q}}(q)}_{q \in E} \text{ is an open cover of } E.
    \]
    Since $E$ is compact, there exists a finite subcover $\set*{N_{r_{q}}(q)}_{q \in S}$ of $E$.
    Let $T$ be a set defined by
    \[
    T \coloneqq \setbuilder*{r_{q}}{q \in S}.
    \]
    Let $f: S \to T$ be a function defined by
    \[
    f(q) = r_{q}.
    \]
    Then $f$ is surjective. By \Cref{thm:SurjectionFromFiniteSetIsFinite}, $T$ is finite. Then, by
    \Cref{thm:FiniteOrderedSetHasExtremum}, $T$ has a minimum element, say $\veps = \min T > 0$. 
    For any $x \in N_{\veps}(p)$, we have
    \[
    d(x, p) < \veps \leq r_{q} \quad \text{for each } q \in S.
    \]
    Assume that there exists some $q \in S$ such that $x \in N_{r_{q}}(q)$. Then we have
    \[
    d(p, q) \leq d(p, x) + d(x, q) < r_{q} + r_{q} = 2r_{q} = d(p, q),
    \]
    which is a contradiction. Then the following statements hold:
    \begin{itemize}
        \item $N_{\veps}(p) \cap N_{r_{q}}(q) = \emptyset$ for each $q \in S$.
        \item $E \subseteq \displaystyle \bigcup_{q \in S} N_{r_{q}}(q)$.
    \end{itemize}
    Therefore, we have $N_{\veps}(p) \subseteq E^{c}$. Since $p$ is arbitrary, $E^{c}$ is open.
\end{proof}

\begin{theorem} \label{thm:ClosedSubsetOfCompactSetIsCompact}
    Let $(X, d)$ be a metric space and $E \subseteq X$. If $E$ is compact, then any closed subset $F \subseteq E$ is compact.
\end{theorem}

\begin{proof}
    Let $\set*{H_{\alpha}}_{\alpha \in A}$ be an open cover of $F$. We have the following statements:
    \begin{itemize}
        \item $F$ is closed. $\implies F^{c}$ is open.
        \item $F \subseteq \displaystyle \bigcup_{\alpha \in A} H_{\alpha}$.
    \end{itemize}
    Then we have
    \begin{itemize}
        \item $X = F \cup F^{c} \subseteq \displaystyle \paren*{\bigcup_{\alpha \in A}
        H_{\alpha}} \cup F^{c} = \bigcup_{\alpha \in A} \paren*{H_{\alpha} \cup F^{c}}$.
        \item $E \subseteq X \implies E \subseteq \displaystyle \bigcup_{\alpha \in A} \paren*{H_{\alpha} \cup F^{c}}$.
    \end{itemize}
    By \Cref{thm:ArbitraryUnionOfOpenSetsIsOpen}, $H_{\alpha} \cup F^{c}$ is open for each $\alpha \in A$. Thus
    \[
    \set*{H_{\alpha} \cup F^{c}}_{\alpha \in A} \text{ is an open cover of } E.
    \]
    Since $E$ is compact, there exists a finite subcover $\set*{H_{\alpha} \cup F^{c}}_{\alpha \in B}$ of $E$. Then we have
    \begin{itemize}
        \item $E \subseteq \displaystyle \bigcup_{\alpha \in B} \paren*{H_{\alpha} \cup F^{c}}$.
        \item $F \subseteq E \implies F \subseteq \displaystyle \bigcup_{\alpha \in B}
        \paren*{H_{\alpha} \cup F^{c}} = \paren*{\bigcup_{\alpha \in B} H_{\alpha}} \cup F^{c}$.
    \end{itemize}
    Since $F \cap F^{c} = \emptyset$, we have
    \[
    F \subseteq \bigcup_{\alpha \in B} H_{\alpha}.
    \]
    Thus $\set*{H_{\alpha}}_{\alpha \in B}$ is a finite subcover of $F$. Therefore, $F$ is compact.
\end{proof}

\begin{theorem} \label{thm:FiniteCompactIntersectionProperty}
    Let $(X, d)$ be a metric space and let $\set*{K_{\alpha}}_{\alpha \in A}$ be a collection of compact sets. Then we have
    \[
    \bigcap_{\alpha \in B} K_{\alpha} \neq \emptyset \quad \text{for any finite subset }
    B \subseteq A \implies \bigcap_{\alpha \in A} K_{\alpha} \neq \emptyset.
    \]
\end{theorem}

\begin{proof}
    Assume that $\displaystyle \bigcap_{\alpha \in A} K_{\alpha} = \emptyset$. Then we have
    \[
    X = \paren*{\bigcap_{\alpha \in A} K_{\alpha}}^{c} = \bigcup_{\alpha \in A} \paren*{K_{\alpha}}^{c}.
    \]
    Since each $K_{\alpha}$ is compact, $K_{\alpha}$ is closed by \Cref{thm:CompactSetIsClosed}. Then we have
    \[
    \text{Each } \paren*{K_{\alpha}}^{c} \text{ is open} \implies
    \set*{\paren*{K_{\alpha}}^{c}}_{\alpha \in A} \text{ is an open cover of } X.
    \]
    Since $K_{\alpha} \subseteq X \subseteq \displaystyle \bigcup_{\beta \in A} \paren*{K_{\beta}}^{c}$, we have
    \[
    \set{\paren*{K_{\beta}}^{c}}_{\beta \in A} \text{ is an open cover of } K_{\alpha}.
    \]
    Since $K_{\alpha}$ is compact, there exists a finite subcover
    $\set*{\paren*{K_{\beta}}^{c}}_{\beta \in B_{\alpha}}$ of $K_{\alpha}$. Then we have
    \begin{align*}
        K_{\alpha} \subseteq \bigcup_{\beta \in B_{\alpha}} \paren*{K_{\beta}}^{c}
        &= \paren*{\bigcap_{\beta \in B_{\alpha}} K_{\beta}}^{c} \\
        &\implies K_{\alpha} \cap \paren*{\bigcap_{\beta \in B_{\alpha}} K_{\beta}} = \emptyset \\
        &\implies \bigcap_{\beta \in B_{\alpha} \cup \set{\alpha}} K_{\beta} = \emptyset.
    \end{align*}
    Since $\set{\alpha}$ and $B_{\alpha}$ are finite, $B_{\alpha} \cup \set{\alpha}$ is also finite
    by \Cref{thm:FiniteUnionOfFiniteSetsAreFinite}, which contradicts the assumption. Therefore, we have
    \[
    \bigcap_{\alpha \in A} K_{\alpha} \neq \emptyset.
    \]
\end{proof}

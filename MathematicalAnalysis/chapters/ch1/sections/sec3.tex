\begin{section}{Fields and the Real Field}

A set $F$ is called a \textbf{field} if it is equipped with two operations,
\emph{addition} and \emph{multiplication}, and satisfies the following axioms:
\begin{enumerate}[label=\textbf{(\Alph*)}]
    \item Axioms for addition
    \begin{itemize}
        \item $x + y \in F$ for all $x, y \in F$.
        \item $x + y = y + x$ for all $x, y \in F$.
        \item $(x + y) + z = x + (y + z)$ for all $x, y, z \in F$.
        \item There exists an element $0 \in F$ such that $x + 0 = x$ for all $x \in F$.
        \item For each $x \in F$, there exists an element $-x \in F$ such that $x + (-x) = 0$.
    \end{itemize}
    \item Axioms for multiplication
    \begin{itemize}
        \item $x \cdot y \in F$ for all $x, y \in F$.
        \item $x \cdot y = y \cdot x$ for all $x, y \in F$.
        \item $(x \cdot y) \cdot z = x \cdot (y \cdot z)$ for all $x, y, z \in F$.
        \item There exists an element $1 \in F$ with $1 \neq 0$ such that $x \cdot 1 = x$ for all $x \in F$.
        \item For each $x \in F$ with $x \neq 0$, there exists an element $x^{-1} \in F$ such that $x \cdot x^{-1} = 1$.
    \end{itemize}
    \item The distributive law
    \begin{itemize}
        \item $x \cdot (y + z) = x \cdot y + x \cdot z$ for all $x, y, z \in F$.
    \end{itemize}
\end{enumerate}

\bigskip

\begin{remark}
    A set $F$ is called an \textbf{ordered field} if it is both an ordered set and a field,
    and satisfies the following axioms:
    \begin{itemize}
        \item If $x, y, z \in F$ and $x < y$, then $x + z < y + z$.
        \item If $x, y \in F$ and $x > 0$ and $y > 0$, then $x \cdot y > 0$.
    \end{itemize}
\end{remark}

\bigskip

\begin{remark}
    There exists an ordered field $\R$ which has the \hyperref[def:LeastUpperBoundProperty]{least upper bound property}.
    Moreover, $\R$ contains $\Q$ as a subfield, i.e., $\Q \subseteq \R$ and the operations of addition and multiplication
    on $\R$ coincide with those on $\Q$.
    \\[3mm]
    We can construct $\R$ from $\Q$. But for now, we will assume the existence of $\R$ and its properties.
\end{remark}

\bigskip

\begin{theorem} \label{thm:ArchimedeanProperty}
    If $x, y \in \R$ and $x > 0$, then there exists some $n \in \N$ such that $nx > y$.
    This theorem is called the \textbf{Archimedean property} of $\R$.
\end{theorem}

\begin{proof}
    Suppose $x, y \in \R$ and $x > 0$. Let $E \coloneqq \setbuilder{nx}{nx \leq y, \; n \in \N}$.
    If $E = \emptyset$, which implies that $x > y$, $n = 1$ satisfies $nx > y$.
    Otherwise, $E$ is nonempty and bounded above by $y$.
    \\[3mm]
    By the least upper bound property, there exists $\alpha = \sup E \in \R$.
    Then $\alpha - x$ is not an upper bound of $E$. Therefore, there exists some
    $m \in \N$ such that $mx > \alpha - x \implies (m + 1)x > \alpha$. Then $(m + 1)x \notin E$,
    which implies that $(m + 1)x > y$. Therefore, we can take $n = m + 1$.
\end{proof}

\bigskip

\begin{theorem} \label{thm:QIsDenseInR}
    If $x, y \in \R$ and $x < y$, then there exists some $p \in \Q$ such that $x < p < y$.
    This theorem states that $\Q$ is \textbf{dense} in $\R$.
\end{theorem}

\begin{proof}
    Let $\epsilon = y - x > 0$. By \Cref{thm:ArchimedeanProperty}, there exists some $n \in \N$
        such that $\dfrac{1}{n} < \epsilon$. Let $S \coloneqq \setbuilder*{k \in \Z}{\dfrac{k}{n} < x}$.
        By \Cref{thm:BoundedSubsetOfZHasExtremum}, there exists $m = \max{S} \in S$. Then we have
        \[
        x < \frac{m + 1}{n} < x + \frac{1}{n} < x + (y - x) = y.
        \]
        Therefore, we can take $p = \dfrac{m + 1}{n} \in \Q$.
\end{proof}

\bigskip

\begin{theorem} \label{thm:ExistenceOfNthRoots}
    For each $x \in \R^{+}$ and $n \in \N$, there exists a unique $y \in \R^{+}$ such that
    $y^{n} = x$. We denote $y$ by $\sqrt[n]{x}$ or $x^{\frac{1}{n}}$.
\end{theorem}

\begin{proof}
    Let $E \coloneqq \setbuilder{t \in \R^{+}}{t^{n} < x}$. Then $E$ is nonempty since
    $2^{-1} \in E$ if $x\geq 1$, and $x^{2} \in E$ otherwise. Moreover, $E$ is bounded above by $\max\set{x, 1}$.
    By the least upper bound property, there exists $\alpha = \sup E \in \R^{+}$. We will show that $\alpha^{n} = x$.
    \\[6mm]
    For any $a, b \in \R^{+}$ with $a < b$, we have
    \[
    b^{n} - a^{n} = (b - a)\sum_{k = 0}^{n - 1} b^{n - 1 - k} a^{k} < (b - a)nb^{n - 1}.
    \]
    Assume that $\alpha^{n} < x$. By \Cref{thm:QIsDenseInR}, we can choose $h \in \R^{+}$ such that
    \[
    h < \min\set*{1, \; \frac{x - \alpha^{n}}{n(\alpha + 1)^{n - 1}}}.
    \]
    Put $a = \alpha$ and $b = \alpha + h$. Then we have
    \begin{align*}
        &(\alpha + h)^{n} - \alpha^{n} < hn(\alpha + h)^{n - 1} < hn(\alpha + 1)^{n - 1} \\
        \implies &(\alpha + h)^{n} < \alpha^{n} + hn(\alpha + 1)^{n - 1} < \alpha^{n} + (x - \alpha^{n}) = x.
    \end{align*}
    This shows that $\alpha + h \in E$, which contradicts the fact that $\alpha = \sup E$.
    \\[6mm]
    Assume that $\alpha^{n} > x$. By \Cref{thm:QIsDenseInR}, we can choose $h \in \R^{+}$ such that
    \[
    h < \min\set*{1, \; \alpha, \; \frac{\alpha^{n} - x}{n(\alpha + 1)^{n - 1}}}.
    \]
    Put $a = \alpha - h$ and $b = \alpha$. Then we have
    \begin{align*}
        &\alpha^{n} - (\alpha - h)^{n} < hn\alpha^{n - 1} < hn(\alpha + 1)^{n - 1} \\
        \implies &(\alpha - h)^{n} > \alpha^{n} - hn(\alpha + 1)^{n - 1} > \alpha^{n} - (\alpha^{n} - x) = x.
    \end{align*}
    This shows that $\alpha - h$ is an upper bound of $E$, which contradicts the fact that $\alpha = \sup E$.
    \\[6mm]
    Therefore, we have $\alpha^{n} = x$, which guarantees the \emph{existence} of $y$. For any $\alpha, \beta \in \R^{+}$,
    $\alpha \neq \beta \implies \alpha^{n} \neq \beta^{n}$. Thus the \emph{uniqueness} of $y$ is guaranteed.
\end{proof}

\bigskip

\begin{remark}
    \Cref{thm:BoundedSubsetOfZHasExtremum} supposes that $E$ is bounded in $\Z$. By \Cref{thm:ArchimedeanProperty},
    we can always find an integer $k$ such that $k < x$ or $x < k$ for any $x \in \R$. 
    Therefore, \Cref{thm:BoundedSubsetOfZHasExtremum} also holds when $E$ is bounded in $\R$, which is followed by
    the fact that $E$ is also bounded in $\Z$.
\end{remark}

\bigskip

\begin{definition} \label{def:FloorCeilingFunctions}
    For any $x \in \R$, the \textbf{floor function} $\floor{x}$
    and \textbf{ceiling function} $\ceil{x}$ are defined as follows:
    \[
    \floor{x} \coloneqq \max\setbuilder{k \in \Z}{k \leq x}, \quad
    \ceil{x} \coloneqq \min\setbuilder{k \in \Z}{k \geq x}
    \]
    whose existences are guaranteed by \Cref{thm:BoundedSubsetOfZHasExtremum}.
\end{definition}

\bigskip

\begin{theorem} \label{thm:DecimalExpansionOfRealNumbers}
    Let $x \in \R$ and $b \in \N$ be given with $b \geq 2$. For any $n \in \N$,
    there exists a unique sequence of integers $\set{a_{1}, a_{2}, \dots, a_{n}}$ such that
    \[
    z_{0} \leq x - \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} < z_{0} + \frac{1}{b^{n}}
    \]
    where $z_{0} = \floor{x}$ and $a_{k} \in S = \set{0, 1, 2, \dots, b - 1}$ for all $k = 1, 2, \dots, n$.
    The sequence $\set{z_{0}, a_{1}, a_{2}, \dots, a_{n}}$ is called the \textbf{base $b$ expansion} of $x$.
    Especially, if $b = 10$, then the sequence is called the \textbf{decimal expansion} of $x$.
\end{theorem}

\begin{proof}
    Let $T_{n} \coloneqq \setbuilder*{k \in S}{z_{n} + \dfrac{k}{b} \leq b^{n}x}$ for $n \in \N \cup \set{0}$,
    where $z_{n} = \floor{b^{n}x}$. Since $0 \in T_{n}$ and $T_{n}$ is bounded above,
    by \Cref{thm:BoundedSubsetOfZHasExtremum}, there exists $a_{n} = \max{T_{n - 1}} \in T_{n - 1}$ for each $n \in \N$.
    \\[3mm]
    First, we will show the \emph{existence} by mathematical induction on $n$.
    Let $P_{n}:$ The sequence of integers $\set{a_{1}, a_{2}, \dots, a_{n}}$ satisfies
    \[
    z_{0} \leq x - \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} < z_{0} + \frac{1}{b^{n}}.
    \]
    \begin{itemize}
        \item \textbf{Base case:}
        Since $a_{1} = \max{T_{0}} \in T_{0}$, we have
        \[
        z_{0} + \frac{a_{1}}{b} \leq x < z_{0} + \frac{a_{1} + 1}{b}
        \implies z_{0} \leq x - \frac{a_{1}
        }{b} < z_{0} + \frac{1}{b}.
        \]
        Thus $P_{1}$ is true.
        \item \textbf{Inductive step:}
        Assume that $P_{n}$ is true for some $n \in \N$. Then we have
        \begin{align*}
            &z_{0} \leq x - \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} < z_{0} + \frac{1}{b^{n}} \\
            \implies &z_{0}b^{n} \leq b^{n}x - \sum_{k = 1}^{n} a_{k} b^{n - k} < z_{0}b^{n} + 1 \\
            \implies &z_{0}b^{n} + \sum_{k = 1}^{n} a_{k} b^{n - k} \leq b^{n}x
            < z_{0}b^{n} + \sum_{k = 1}^{n} a_{k} b^{n - k} + 1.
        \end{align*}
        By \Cref{def:FloorCeilingFunctions}, we have
        \[
        z_{n} = z_{0}b^{n} + \sum_{k = 1}^{n} a_{k} b^{n - k}.
        \]
        Since $a_{n + 1} = \max{T_{n}} \in T_{n}$, we have
        \begin{align*}
            &z_{n} + \frac{a_{n + 1}}{b} \leq b^{n}x < z_{n} + \frac{a_{n + 1} + 1}{b} \\
            \implies &z_{0}b^{n} + \sum_{k = 1}^{n} a_{k} b^{n - k} + \frac{a_{n + 1}}{b}
            \leq b^{n}x < z_{0}b^{n} + \sum_{k = 1}^{n} a_{k} b^{n - k} + \frac{a_{n + 1} + 1}{b} \\
            \implies &z_{0}b^{n} + \sum_{k = 1}^{n + 1} a_{k} b^{n - k} \leq b^{n}x
            < z_{0}b^{n} + \sum_{k = 1}^{n + 1} a_{k} b^{n - k} + \frac{1}{b} \\
            \implies &z_{0} + \sum_{k = 1}^{n + 1} \frac{a_{k}}{b^{k}} \leq x
            < z_{0} + \sum_{k = 1}^{n + 1} \frac{a_{k}}{b^{k}} + \frac{1}{b^{n + 1}} \\
            \implies &z_{0} \leq x - \sum_{k = 1}^{n + 1} \frac{a_{k}}{b^{k}} < z_{0} + \frac{1}{b^{n + 1}}.
        \end{align*}
        Thus $P_{n + 1}$ is true whenever $P_{n}$ is true.
    \end{itemize}
    By the \hyperref[rem:PrincipleOfMathematicalInduction]{principle of mathematical induction}, $P_{n}$ is true for all
    $n \in \N$. Thus the \emph{existence} of the sequence $\set{a_{1}, a_{2}, \dots, a_{n}}$ for each $n \in \N$ is guaranteed.
    \\[6mm]
    Next, we will show the \emph{uniqueness} of the sequence. Suppose that there exist two sequences of integers
    $\set{a_{1}, a_{2}, \dots, a_{n}}$ and $\set{c_{1}, c_{2}, \dots, c_{n}}$ such that
    \begin{itemize}
        \item $\displaystyle z_{0} \leq x - \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} < z_{0} + \frac{1}{b^{n}}$.
        \item $\displaystyle z_{0} \leq x - \sum_{k = 1}^{n} \frac{c_{k}}{b^{k}} < z_{0} + \frac{1}{b^{n}}$.
    \end{itemize}
    for some $n \in \N$. Then we have both
    \begin{itemize}
        \item $\displaystyle \sum_{k = 1}^{n} \dfrac{a_{k}}{b^{k}} \leq x - z_{0} <
        \sum_{k = 1}^{n} \dfrac{c_{k}}{b^{k}} + \dfrac{1}{b^{n}}$.
        \item $\displaystyle \sum_{k = 1}^{n} \dfrac{c_{k}}{b^{k}} \leq x - z_{0} <
        \sum_{k = 1}^{n} \dfrac{a_{k}}{b^{k}} + \dfrac{1}{b^{n}}$.
    \end{itemize}
    By combining the two inequalities, we have
    \begin{align*}
        \sum_{k = 1}^{n} \abs*{\frac{a_{k} - c_{k}}{b^{k}}} < \frac{1}{b^{n}} \implies
        &\abs*{\frac{a_{k} - c_{k}}{b^{k}}} < \frac{1}{b^{n}} \text{ for all } k = 1, 2, \dots, n \\
        \implies &\abs{a_{k} - c_{k}} < b^{k - n} \leq 1 \text{ for all } k = 1, 2, \dots, n.
    \end{align*}
    Since $a_{k}, c_{k} \in S$ and so is $\abs{a_{k} - c_{k}}$, we have $a_{k} = c_{k}$ for all $k = 1, 2, \dots, n$.
    Thus the \emph{uniqueness} of the sequence $\set{a_{1}, a_{2}, \dots, a_{n}}$ for each $n \in \N$ is guaranteed.
\end{proof}

    By the definition of $a_{n}$, we have
    \begin{align*}
        &a_{n} \leq b^{n}y - \sum_{k = 1}^{n - 1} a_{k}b^{n - k} < a_{n} + 1 \\
        \implies &0 \leq b^{n}y - \sum_{k = 1}^{n} a_{k}b^{n - k} < 1 \\
        \implies &0 \leq y - \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} < \frac{1}{b^{n}}
    \end{align*}
    for each $n \in \N$. Therefore, the \emph{existence} of the \emph{greedy base $b$ expansion} of $y$ is guaranteed.
    \\[6mm]
    Next, assume that there exist two sequences $\paren*{a_{n}}_{n = 1}^{\infty}$
    and $\paren*{c_{n}}_{n = 1}^{\infty}$ such that for each $n \in \N$, we have

    \begin{itemize}
        \item $\displaystyle 0 \leq y - \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} < \frac{1}{b^{n}}$
        \item $\displaystyle 0 \leq y - \sum_{k = 1}^{n} \frac{c_{k}}{b^{k}} < \frac{1}{b^{n}}$
    \end{itemize}
    where $a_{k} \in S_{b}$ and $c_{k} \in S_{b}$ for all $k = 1, 2, \dots, n$. We will show that
    for each $n \in \N$, we have $a_{n} = c_{n}$. Let $Q(n)$ be the statement that
    \[
    Q(n) : a_{k} = c_{k} \text{ for all } k = 1, 2, \dots, n.
    \]
    \begin{itemize}
        \item \textbf{Base case:} We have
        \begin{itemize}
            \item $0 \leq y - \dfrac{a_{1}}{b} < \dfrac{1}{b} \implies \dfrac{a_{1}}{b} \leq y < \dfrac{a_{1} + 1}{b}$.
            \item $0 \leq y - \dfrac{c_{1}}{b} < \dfrac{1}{b} \implies \dfrac{c_{1}}{b} \leq y < \dfrac{c_{1} + 1}{b}$. 
        \end{itemize}
        By combining the two inequalities, we have
        \begin{itemize}
            \item $\dfrac{a_{1}}{b} \leq y < \dfrac{c_{1} + 1}{b} \implies a_{1} < c_{1} + 1 \implies a_{1} \leq c_{1}$.
            \item $\dfrac{c_{1}}{b} \leq y < \dfrac{a_{1} + 1}{b} \implies c_{1} < a_{1} + 1 \implies c_{1} \leq a_{1}$.
        \end{itemize}
        Therefore, we have $a_{1} = c_{1}$, which shows that $Q(1)$ is true.
        \item \textbf{Inductive step:} Assume that $Q(n)$ is true for some $n \in \N$. We have
        \begin{itemize}
            \item $\displaystyle 0 \leq y - \sum_{k = 1}^{n + 1} \frac{a_{k}}{b^{k}} < \frac{1}{b^{n + 1}} \implies
            \sum_{k = 1}^{n + 1} \frac{a_{k}}{b^{k}} \leq y < \sum_{k = 1}^{n + 1} \frac{a_{k}}{b^{k}} + \frac{1}{b^{n + 1}}$.
            \item $\displaystyle 0 \leq y - \sum_{k = 1}^{n + 1} \frac{c_{k}}{b^{k}} < \frac{1}{b^{n + 1}} \implies
            \sum_{k = 1}^{n + 1} \frac{c_{k}}{b^{k}} \leq y < \sum_{k = 1}^{n + 1} \frac{c_{k}}{b^{k}} + \frac{1}{b^{n + 1}}$.
        \end{itemize}
        By combining the two inequalities, we have
        \begin{itemize}
            \item $\displaystyle \sum_{k = 1}^{n + 1} \frac{a_{k}}{b^{k}} \leq y < \sum_{k = 1}^{n + 1} \frac{c_{k}}{b^{k}}
            + \frac{1}{b^{n + 1}} \implies \sum_{k = 1}^{n + 1} \frac{a_{k} - c_{k}}{b^{k}} < \frac{1}{b^{n + 1}}$.
            \item $\displaystyle \sum_{k = 1}^{n + 1} \frac{c_{k}}{b^{k}} \leq y < \sum_{k = 1}^{n + 1} \frac{a_{k}}{b^{k}}
            + \frac{1}{b^{n + 1}} \implies \sum_{k = 1}^{n + 1} \frac{c_{k} - a_{k}}{b^{k}} < \frac{1}{b^{n + 1}}$.
        \end{itemize}
        By the inductive hypothesis, we have $a_{k} = c_{k}$ for all $k = 1, 2, \dots, n$. Therefore, we have
        \begin{itemize}
            \item $\displaystyle \frac{a_{n + 1} - c_{n + 1}}{b^{n + 1}} < \frac{1}{b^{n + 1}}
            \implies a_{n + 1} - c_{n + 1} < 1 \implies a_{n + 1} \leq c_{n + 1}$.
            \item $\displaystyle \frac{c_{n + 1} - a_{n + 1}}{b^{n + 1}} < \frac{1}{b^{n + 1}}
            \implies c_{n + 1} - a_{n + 1} < 1 \implies c_{n + 1} \leq a_{n + 1}$.
        \end{itemize}
        Therefore, we have $a_{n + 1} = c_{n + 1}$, which shows that $Q(n + 1)$ is true whenever $Q(n)$ is true.
    \end{itemize}
    By the \emph{principle of mathematical induction}, $Q(n)$ is true for all
    $n \in \N$. Therefore, the \emph{uniqueness} of the \emph{greedy base $b$ expansion} of $y$ is guaranteed.
\end{proof}

\begin{theorem} \label{thm:SameGreedyBaseExpansionImpliesEquality}
    Let $b \in \N$ be given with $b \geq 2$. Let $\paren*{a_{n}}_{n = 1}^{\infty}$ be the greedy base $b$ expansion of
    $a \in [0, 1)$, and let $\paren*{c_{n}}_{n = 1}^{\infty}$ be the greedy base $b$ expansion of $c \in [0, 1)$.
    If $a_{n} = c_{n}$ for all $n \in \N$, then we have $a = c$.
\end{theorem}

\begin{proof}
    Assume that $a \neq c$. Without loss of generality, assume that $a < c$. Then,
    by \Cref{cor:ArchimedeanPropertyCorollary}, there exists some $n \in \N$ such that
    \[
    \frac{1}{b^{n}} < c - a.
    \]
    By the definition of the \emph{greedy base $b$ expansion}, we have
    \begin{itemize}
        \item $\displaystyle 0 \leq a - \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} < \frac{1}{b^{n}}
        \implies \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} + \frac{1}{b^{n}} \leq a + \frac{1}{b^{n}}$.
        \item $\displaystyle 0 \leq c - \sum_{k = 1}^{n} \frac{c_{k}}{b^{k}} < \frac{1}{b^{n}}
        \implies c < \sum_{k = 1}^{n} \frac{c_{k}}{b^{k}} + \frac{1}{b^{n}}$.
    \end{itemize}
    By the assumption that $a_{n} = c_{n}$ for all $n \in \N$, we have
    \[
    \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} = \sum_{k = 1}^{n} \frac{c_{k}}{b^{k}}.
    \]
    Therefore, we have a contradiction:
    \[
    \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} + \frac{1}{b^{n}} \leq a + \frac{1}{b^{n}}
    < c < \sum_{k = 1}^{n} \frac{c_{k}}{b^{k}} + \frac{1}{b^{n}}.
    \]
    Therefore, we have $a = c$, which completes the proof.
\end{proof}

\begin{theorem} \label{thm:NoRealNumberMatchingGreedyBaseExpansion}
    Let $b \in \N$ be given with $b \geq 2$, and let $\paren*{a_{n}}_{n = 1}^{\infty}$ be given with $a_{k} \in S_{b}$
    for all $k \in \N$. If there exists no $y \in [0, 1)$ such that
    \[
    \paren*{a_{n}}_{n = 1}^{\infty} \text{ is the greedy base $b$ expansion of $y$},
    \]
    then there exists some $N \in \N$ such that
    \[
    a_{n} = b - 1 \quad \text{for all } n \geq N.
    \]
\end{theorem}

\begin{proof}
    We will prove the contrapositive of the statement: For all $N \in \N$, if there exists some $n \in \N$ such that
    \[
    n \geq N \text{ and } a_{n} \neq b - 1,
    \]
    then there exists $y \in [0, 1)$ such that $\paren*{a_{n}}_{n = 1}^{\infty}$ is the \emph{greedy base $b$ expansion} of $y$.
    \\[6mm]
    Assume that the condition of the above claim holds. Let $x_{n} \coloneqq \displaystyle \sum_{k = 1}^{n}
    \frac{a_{k}}{b^{k}}$ for each $n \in \N$. Since $a_{k} \in S_{b}$ for all $k \in \N$, we have
    \[
    0 \leq x_{n} \leq \sum_{k = 1}^{n} \frac{b - 1}{b^{k}} = 1 - \frac{1}{b^{n}} < 1 \quad \text{for all } n \in \N.
    \]
    Then, we have that $X \coloneqq \setbuilder*{x_{n}}{n \in \N}$ is bounded above. By the
    \emph{least upper bound property}, there exists $y = \sup X \geq 0$.
    \\[6mm]
    First, we will show that $y < 1$. By the assumption, there exists $m \in \N$ such that $a_{m} \neq b - 1$.
    Then, for all $n > m$, we have
    \begin{align*}
        &x_{n} = x_{m} + \sum_{k = m + 1}^{n} \frac{a_{k}}{b^{k}} \leq x_{m} + \sum_{k = m + 1}^{n} \frac{b - 1}{b^{k}} \\
        \implies &x_{n} \leq x_{m} + \frac{1}{b^{m}} - \frac{1}{b^{n}} < x_{m} + \frac{1}{b^{m}}.
    \end{align*}
    Since $a_{m} \leq b - 2$, we have
    \begin{align*}
        x_{m} + \frac{1}{b^{m}} &= \sum_{k = 1}^{m - 1} \frac{a_{k}}{b^{k}} + \frac{a_{m} + 1}{b^{m}} \\
        &\leq \sum_{k = 1}^{m - 1} \frac{a_{k}}{b^{k}} + \frac{b - 1}{b^{m}} \\
        &\leq \sum_{k = 1}^{m} \frac{b - 1}{b^{k}} = 1 - \frac{1}{b^{m}}.
    \end{align*}
    We have $x_{n} \leq x_{m}$ for all $n \leq m$, and $x_{n} < x_{m} + \dfrac{1}{b^{m}} \leq 1 - \dfrac{1}{b^{m}}$
    for all $n > m$. Thus $1 - \dfrac{1}{b^{m}}$ is an upper bound of $X$. Since $y = \sup X$, we have
    \[
    y \leq 1 - \frac{1}{b^{m}} < 1 \implies y \in [0, 1).
    \]
    \\[6mm]
    Next, we will show that $\paren*{a_{n}}_{n = 1}^{\infty}$ is the \emph{greedy base $b$ expansion} of $y$.
    Let $n \in \N$ be fixed. By the assumption, there exists $m \in \N$ such that $m \geq n$ and $a_{m} \leq b - 2$.
    \begin{align*}
        &x_{m} = x_{n} + \sum_{k = n + 1}^{m} \frac{a_{k}}{b^{k}} < x_{n} + \sum_{k = n + 1}^{m} \frac{b - 1}{b^{k}} \\
        \implies &x_{m} < x_{n} + \frac{1}{b^{n}} - \frac{1}{b^{m}}
        \implies x_{m} + \frac{1}{b^{m}} < x_{n} + \frac{1}{b^{n}}.
    \end{align*}
    Since $a_{m} \leq b - 2$, $x_{m} + \dfrac{1}{b^{m}}$ is an upper bound of $X$ by the same argument as above.
    Since $y = \sup X$, we have
    \[
    y \leq x_{m} + \frac{1}{b^{m}} < x_{n} + \frac{1}{b^{n}} \implies 0 \leq y - x_{n} < \frac{1}{b^{n}}.
    \]
    Therefore, $\paren*{a_{n}}_{n = 1}^{\infty}$ is the \emph{greedy base $b$ expansion} of $y$.
\end{proof}

\begin{theorem} \label{thm:NoRealNumberMatchingGreedyBaseExpansionConverse}
    Let $b \in \N$ be given with $b \geq 2$ and let $\paren*{a_{n}}_{n = 1}^{\infty}$ be given with $a_{k} \in S_{b}$
    for all $k \in \N$. If there exists some $N \in \N$ such that
    \[
    a_{n} = b - 1 \quad \text{for all } n \geq N,
    \]
    then there exists no $y \in [0, 1)$ such that $\paren*{a_{n}}_{n = 1}^{\infty}$ is the greedy base $b$ expansion of $y$.
    This theorem is the converse of \Cref{thm:NoRealNumberMatchingGreedyBaseExpansion}.
\end{theorem}

\begin{proof}
    Assume that there exists some $y \in [0, 1)$ such that $\paren*{a_{n}}_{n = 1}^{\infty}$
    is the \emph{greedy base $b$ expansion} of $y$. Then we have
    \[
    0 \leq y - \sum_{k = 1}^{n} \frac{a_{k}}{b^{k}} < \frac{1}{b^{n}} \quad \text{for all } n \in \N.
    \]
    Let $N \in \N$ be such that $a_{n} = b - 1$ for all $n \geq N$. Let $x_{n} \coloneqq \displaystyle \sum_{k = 1}^{n}
    \frac{a_{k}}{b^{k}}$ for each $n \in \N$. Then, for all $n \geq N$, we have
    \begin{align*}
        &x_{n} = x_{N - 1} + \sum_{k = N}^{n} \frac{b - 1}{b^{k}} = x_{N - 1} + \frac{1}{b^{N - 1}} - \frac{1}{b^{n}} \\
        \implies &y - x_{n} = y - x_{N - 1} - \frac{1}{b^{N - 1}} + \frac{1}{b^{n}} \geq 0 \\
        \implies &\frac{1}{b^{n}} \geq x_{N - 1} + \frac{1}{b^{N - 1}} - y.
    \end{align*}
    Since $0 \leq y - x_{N - 1} < \dfrac{1}{b^{N - 1}}$, we have $x_{N - 1} + \dfrac{1}{b^{N - 1}} - y > 0$. By
    \Cref{cor:ArchimedeanPropertyCorollary}, there exists some $n \in \N$ with $n \geq N$ such that
    \[
    \frac{1}{b^{n}} < x_{N - 1} + \frac{1}{b^{N - 1}} - y,
    \]
    which is a contradiction. Therefore, there exists no $y \in [0, 1)$ such that
    $\paren*{a_{n}}_{n = 1}^{\infty}$ is the \emph{greedy base $b$ expansion} of $y$.
\end{proof}

\begin{section}{Euclidean Spaces}

For $n \in \N$, we define the \textbf{Euclidean $n$-space} $\R^{n}$
as the set of all ordered $n$-tuples of real numbers, i.e,
\[
\R^{n} \coloneqq \setbuilder*{\bm{x} = (x_{1}, x_{2}, \dots, x_{n})}{x_{k} \in \R \text{ for all } k = 1, 2, \dots, n}
\]
where $x_{1}, x_{2}, \dots, x_{n}$ are called the \textbf{coordinates} of $\bm{x}$.
The elements of $\R^{n}$ with $n > 1$ are called \textbf{vectors} or \emph{points}, which are denoted by boldface letters.
\\[3mm]
Let $\bm{x}, \bm{y} \in \R^{n}$ and $\alpha \in \R$. We define the following operations on $\R^{n}$:
\begin{itemize}
    \item $\bm{x} + \bm{y} = (x_{1} + y_{1}, x_{2} + y_{2}, \dots, x_{n} + y_{n})$.
    \item $\alpha \bm{x} = (\alpha x_{1}, \alpha x_{2}, \dots, \alpha x_{n})$.
\end{itemize}
Then $\R^{n}$ is a \emph{vector space} over $\R$ with the zero vector $\bm{0} = (0, 0, \dots, 0)$.

\bigskip

\begin{definition} \label{def:InnerProductNormDistance}
    Let $\bm{x}, \bm{y} \in \R^{n}$. The \textbf{inner product} of $\bm{x}$ and $\bm{y}$ is defined as
    \[
    \bm{x} \cdot \bm{y} = \sum_{k = 1}^{n} x_{k}y_{k}.
    \]
    The \textbf{norm} of $\bm{x}$ is defined as
    \[
    \norm{\bm{x}} = \sqrt{\bm{x} \cdot \bm{x}} = \sqrt{\sum_{k = 1}^{n} x_{k}^{2}}.
    \]
    The \textbf{distance} between $\bm{x}$ and $\bm{y}$ is defined as
    \[
    d(\bm{x}, \bm{y}) = \norm{\bm{x} - \bm{y}} = \sqrt{\sum_{k = 1}^{n} (x_{k} - y_{k})^{2}}.
    \]
    We can confirm that $d(x, y) = \abs{x - y}$ in the case of $n = 1$.
\end{definition}

\bigskip

\begin{theorem} \label{thm:PropertiesOfEuclideanSpaces}
    Let $\bm{x}, \bm{y}, \bm{z} \in \R^{n}$ and $\alpha \in \R$. Then we have the following properties:
    \begin{enumerate}[label=\textbf{(\Alph*)}]
        \item $\abs{\bm{x} \cdot \bm{y}} \leq \norm{\bm{x}} \cdot \norm{\bm{y}}$.
        \item $\norm{\bm{x} + \bm{y}} \leq \norm{\bm{x}} + \norm{\bm{y}}$.
        \item $\norm{\bm{x} - \bm{y}} \geq \abs{\norm{\bm{x}} - \norm{\bm{y}}}$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    \begin{enumerate}[label=\textbf{(\Alph*)}]
        \item By \Cref{cor:CauchySchwarzInequalityForRealNumbers}, we have
            \[
            \abs{\bm{x} \cdot \bm{y}}^{2} = \abs*{\sum_{k = 1}^{n} x_{k}y_{k}}^{2} \leq
            \sum_{k = 1}^{n} x_{k}^{2} \cdot \sum_{k = 1}^{n} y_{k}^{2}
            = \norm{\bm{x}}^{2} \cdot \norm{\bm{y}}^{2}.
            \]
        \item By part \textbf{(A)}, we have
            \begin{align*}
                \norm{\bm{x} + \bm{y}}^{2} = \; &(\bm{x} + \bm{y}) \cdot (\bm{x} + \bm{y})
                = \norm{\bm{x}}^{2} + 2\bm{x} \cdot \bm{y} + \norm{\bm{y}}^{2} \\
                \leq \; &\norm{\bm{x}}^{2} + 2\abs{\bm{x} \cdot \bm{y}} + \norm{\bm{y}}^{2}
                \leq \norm{\bm{x}}^{2} + 2\norm{\bm{x}} \cdot \norm{\bm{y}} + \norm{\bm{y}}^{2} \\
                = \; &(\norm{\bm{x}} + \norm{\bm{y}})^{2}.
            \end{align*}
        \item By part \textbf{(B)}, we have
            \begin{align*}
                &\norm{\bm{x}} = \norm{(\bm{x} - \bm{y}) + \bm{y}} \leq \norm{\bm{x} - \bm{y}}
                + \norm{\bm{y}} \implies \norm{\bm{x}} - \norm{\bm{y}} \leq \norm{\bm{x} - \bm{y}}. \\
                &\norm{\bm{y}} = \norm{(\bm{y} - \bm{x}) + \bm{x}} \leq \norm{\bm{y} - \bm{x}}
                + \norm{\bm{x}} \implies \norm{\bm{y}} - \norm{\bm{x}} \leq \norm{\bm{x} - \bm{y}}.
            \end{align*}
    \end{enumerate}
\end{proof}

\end{section}

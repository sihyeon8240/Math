\begin{section}{The Natural, Integer, and Rational Number Systems}

We denote the set $\set*{1, 2, 3, \dots}$ of natural numbers by $\N$.
The set $\N$ is defined by the following axioms:
\begin{itemize}
    \item $1 \in \N$
    \item $n \in \N \implies s(n) \in \N$
    \item $\nexists n \in \N$ such that $s(n) = 1$
    \item $m, n \in \N, s(m) = s(n) \implies m = n$
    \item If $S \subseteq \N$ satisfies
    \begin{itemize}
        \item $1 \in S$
        \item $n \in S \implies s(n) \in S$
    \end{itemize}
    then $S = \N$
\end{itemize}
$s(n) \coloneqq n + 1$ is called the \textbf{successor} of $n$.
These axioms are called the \textbf{Peano axioms} (or \emph{Peano Postulates}).
\\[6mm]
We define the set $\Z$ of integers and the set $\Q$ of rational numbers as follows:
\begin{itemize}
    \item $\Z \coloneqq \setbuilder*{m - n}{m, n \in \N}$
    \item $\Q \coloneqq \setbuilder*{\dfrac{m}{n}}{m \in \Z, \; n \in \N}$
\end{itemize}
Also, we denote the set of positive numbers or negative numbers as follows:
\begin{itemize}
    \item $\Z^{+} \coloneqq \setbuilder*{n}{n \in \N}$, $\Z^{-} \coloneqq \setbuilder*{-n}{n \in \N}$
    \item $\Q^{+} \coloneqq \setbuilder*{\dfrac{m}{n}}{m \in \N, \; n \in \N}$,
    $\Q^{-} \coloneqq \setbuilder*{\dfrac{m}{n}}{m \in \Z^{-}, \; n \in \N}$
\end{itemize}

\bigskip

\begin{remark}
    The \textbf{Peano axioms} do not guarantee the \emph{existence} of $\N$, which is guaranteed by the
    \textbf{axiom of infinity} in the \textbf{ZFC axioms}.
    \\[6mm]
    In this textbook, the \textbf{ZFC axioms} will not be discussed, so we will just assume that $\N$ exists.
    Also, we will reject the \textbf{axiom of choice} and provide constructive proofs as far as possible.
\end{remark}

\bigskip

\begin{remark}
    Let $P(n)$ be a statement about $n \in \N$, and let
    \[
    S \coloneqq \setbuilder*{n \in \N}{P(n) \text{ is true}}
    \]
    Assume that $P(n)$ satisfies
    \begin{itemize}
        \item $P(1)$ is true
        \item $P(n)$ is true $\implies P(n + 1)$ is true
    \end{itemize}
    This implies that
    \begin{itemize}
        \item $1 \in S$
        \item $n \in S \implies n + 1 \in S$
    \end{itemize}
    Therefore,
    \[
    \begin{array}{c}
        S = \N \\
        \Downarrow \\
        P(n) \text{ is true for all } n \in \N
    \end{array}
    \]
    This method of proof is called the \textbf{principle of mathematical induction}.
\end{remark}

\bigskip

\begin{theorem} \label{thm:WellOrderingProperty}
    Any nonempty subset of $\N$ has a least element. This theorem is called the \textbf{well-ordering property} of $\N$.
\end{theorem}

\begin{proof}
    Let $S \subseteq \N$ be nonempty. Assume that $S$ has no least element. Let
    \begin{itemize}
        \item $T \coloneqq \N \setminus S$
        \item $L \coloneqq \setbuilder*{n \in \N}{\set*{1, 2, \dots, n} \subseteq T}$
    \end{itemize}
    We will show that
    \[
    \begin{array}{c}
        L = \N \\
        \Downarrow \\
        T = \N \\
        \Downarrow \\
        S = \emptyset \\
        \Downarrow \\
        \bot
    \end{array}
    \]
    \begin{itemize}
        \item \textbf{Base case:} If $1 \in S$, then
        \[
        \begin{array}{c}
            1 \in S \\
            \Downarrow \\
            1 = \min S \\
            \Downarrow \\
            \bot
        \end{array}
        \]
        Therefore,
        \[
        \begin{array}{c}
            1 \notin S \\
            \Downarrow \\
            1 \in T \\
            \Downarrow \\
            1 \in L
        \end{array}
        \]
        \item \textbf{Inductive step:} Assume that $n \in L$ for some $n \in \N$. Then
        \[
        \begin{array}{c}
            n \in L \\
            \Downarrow \\
            \set*{1, 2, \dots, n} \subseteq T \\
            \Downarrow \\
            \set*{1, 2, \dots, n} \cap S = \emptyset
        \end{array}
        \]
        If $n + 1 \in S$, then 
        \[
        \begin{array}{c}
            n + 1 \in S \\
            \Downarrow \\
            n + 1 = \min S \\
            \Downarrow \\
            \bot
        \end{array}
        \]
        Therefore,
        \[
        \begin{array}{c}
            n + 1 \notin S \\
            \Downarrow \\
            n + 1 \in T \\
            \Downarrow \\
            \set*{1, 2, \dots, n + 1} \subseteq T \\
            \Downarrow \\
            n + 1 \in L
        \end{array}
        \]
    \end{itemize}
    By the \emph{principle of mathematical induction}, $L = \N$.
\end{proof}

\end{section}

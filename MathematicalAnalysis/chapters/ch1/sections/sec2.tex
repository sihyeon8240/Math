\begin{section}{Relations}

Let $S$ be a set. A \textbf{relation} $\sim$ on $S$ is a subset of $S \times S$.
Equivalently, a subset $R$ of $S \times S$ is a relation on $S$.
\begin{itemize}
    \item $R = \setbuilder*{(x, y) \in S \times S}{x \sim y}$
    \item $x \sim y \iff (x, y) \in R$
\end{itemize}

\bigskip

\begin{definition} \label{def:EquivalenceRelation}
    Let $S$ be a set. A relation $\sim$ on $S$ is called an \textbf{equivalence relation} if it satisfies the following properties:
    \begin{itemize}
        \item \textbf{Reflexivity}: $\forall x \in S, x \sim x$
        \item \textbf{Symmetry}: $\forall x, y \in S, x \sim y \implies y \sim x$
        \item \textbf{Transitivity}: $\forall x, y, z \in S, \paren*{x \sim y \land y \sim z} \implies x \sim z$
    \end{itemize}
\end{definition}

\bigskip

\begin{definition} \label{def:Partition}
    Let $S$ be a set. A collection $\set*{A_{\alpha}}_{\alpha \in I}$ of sets is called a \textbf{partition} of $S$ if
    \begin{itemize}
        \item $\forall \alpha \in I, \paren*{A_{\alpha} \neq \vnot \land A_{\alpha} \subseteq S}$
        \item $\forall \alpha, \beta \in I, \paren*{\alpha \neq \beta \implies A_{\alpha} \cap A_{\beta} = \vnot}$
        \item $\displaystyle \bigcup_{\alpha \in I} A_{\alpha} = S$
    \end{itemize}
\end{definition}

\bigskip

\begin{definition} \label{def:EquivalenceClass}
    Let $S$ be a set, and let $\sim$ be an equivalence relation on $S$. For each $x \in S$,
    the \textbf{equivalence class} of $x$ is defined by
    \[
    [x] \coloneqq \setbuilder*{y \in S}{y \sim x}
    \]
    We denote the set of all equivalence classes of $S$ by
    \[
    S / \sim \; \coloneqq \setbuilder*{[x]}{x \in S}
    \]
\end{definition}

\bigskip

\begin{theorem} \label{thm:EquivalenceClassesFormAPartition}
    Let $S$ be a set, and let $\sim$ be an equivalence relation on $S$. Then
    \[
    \forall x, y \in S, [x] \cap [y] \neq \vnot \implies [x] = [y]
    \]
\end{theorem}

\begin{proof}
    Assume that $[x] \cap [y] \neq \vnot$. Then
    \[
    \begin{array}{c}
        \exists z \in S, z \in [x] \cap [y] \\
        \Downarrow \\
        z \in [x] \land z \in [y] \\
        \Downarrow \\
        z \sim x \land z \sim y \\
        \Downarrow \\
        x \sim z \land z \sim y \\
        \Downarrow \\
        x \sim y
    \end{array}
    \]
    Therefore, for any $w \in [x]$,
    \[
    \begin{array}{c}
        w \sim x \land x \sim y \\
        \Downarrow \\
        w \sim y \\
        \Downarrow \\
        w \in [y]
    \end{array}
    \]
    Therefore,
    \[
    [x] \subseteq [y]
    \]
    Similarly,
    \[
    [y] \subseteq [x]
    \]
\end{proof}

\bigskip

\begin{theorem} \label{thm:PartitionDefinesEquivalenceRelation}
    Let $S$ be a set, and let $\set*{A_{\alpha}}_{\alpha \in I}$ be a partition of $S$. Let
    \[
    x \sim y \iff \exists \alpha \in I, \paren*{x \in A_{\alpha} \land y \in A_{\alpha}}
    \]
    for all $x, y \in S$.Then
    \begin{enumerate}[label=\textbf{(\Alph*)}]
        \item $\sim$ is an equivalence relation on $S$
        \item $\set*{A_{\alpha}}_{\alpha \in I} = S / \sim$
    \end{enumerate}
\end{theorem}

\begin{proof}
    \begin{enumerate}[label=\textbf{(\Alph*)}]
        \item Let $x, y, z \in S$.
        \begin{itemize}
            \item \textbf{Reflexivity}: For any $x$,
            \[
            \begin{array}{c}
            x \in S \\
            \Downarrow \\
            x \in \displaystyle \bigcup_{\alpha \in I} A_{\alpha} \\
            \Downarrow \\
            \exists \alpha \in I, x \in A_{\alpha} \\
            \Downarrow \\
            x \sim x
            \end{array}
            \]
            \item \textbf{Symmetry}: If $x \sim y$, then
            \[
            \begin{array}{c}
            \exists \alpha \in I, \paren*{x \in A_{\alpha} \land y \in A_{\alpha}} \\
            \Downarrow \\
            y \in A_{\alpha} \land x \in A_{\alpha} \\
            \Downarrow \\
            y \sim x
            \end{array}
            \]
            \item \textbf{Transitivity}: If $x \sim y \land y \sim z$, then
            \[
            \begin{array}{c}
            \exists \alpha \in I, \paren*{x \in A_{\alpha} \land y \in A_{\alpha}} \\
            \exists \beta \in I, \paren*{y \in A_{\beta} \land z \in A_{\beta}} \\
            \Downarrow \\
            y \in A_{\alpha} \land y \in A_{\beta} \\
            \Downarrow \\
            A_{\alpha} \cap A_{\beta} \neq \vnot \\
            \Downarrow \\
            \alpha = \beta \\
            \Downarrow \\
            x \in A_{\alpha} \land z \in A_{\alpha} \\
            \Downarrow \\
            x \sim z
            \end{array}
            \]
        \end{itemize}
        \item Let $x \in S$. Then
        \[
        \begin{array}{c}
            x \in S \\
            \Downarrow \\
            x \in \displaystyle \bigcup_{\alpha \in I} A_{\alpha} \\
            \Downarrow \\
            \exists \alpha \in I, x \in A_{\alpha} \\
            \Downarrow \\
            x \in [x] \\
            \Downarrow \\
            \exists \alpha \in I, A_{\alpha} = [x]
        \end{array}
        \]
        Therefore,
        \[
        \set*{A_{\alpha}}_{\alpha \in I} \subseteq S / \sim
        \]
        Let $[x] \in S / \sim$. Then
        \[
        \begin{array}{c}
            x \in S \\
            \Downarrow \\
            x \in \displaystyle \bigcup_{\alpha \in I} A_{\alpha} \\
            \Downarrow \\
            \exists \alpha \in I, x \in A_{\alpha} \\
            \Downarrow \\
            A_{\alpha} \cap [x] \neq \vnot \\
            \Downarrow \\
            A_{\alpha} = [x]
        \end{array}
        \]
        Therefore,
        \[
        S / \sim \subseteq \set*{A_{\alpha}}_{\alpha \in I}
        \]
    \end{enumerate}
\end{proof}

\end{section}

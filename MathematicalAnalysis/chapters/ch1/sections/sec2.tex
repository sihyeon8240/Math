\begin{section}{Ordered Sets and Ordered Fields}

A set $S$ is called an \textbf{ordered set} if it is equipped with a relation $<$
such that for any $x, y, z \in S$, we have the following properties:
\begin{enumerate}[label=\textbf{(\Alph*)}]
    \item $x < y$ and $y < z$ implies $x < z$.
    \item Only one of the relations $x < y$, $x = y$, or $x > y$ holds.
\end{enumerate}
$x > y$ means $y < x$, $x \leq y$ means ($x < y$ or $x = y$) and $x \geq y$ means $y \leq x$.

\bigskip

\begin{remark}
    In an ordered set $S$, the relation $<$ is called an \textbf{order} (or a \textbf{total order}) on $S$.
    If a relation $<$ on $T$ satisfies \textbf{(A)} only, then the relation $<$ is called a
    \textbf{partial order} on $T$ and the set $T$ is called \textbf{partially ordered}.
\end{remark}

\bigskip

A set $F$ is called a \textbf{field} if it is equipped with two operations,
\emph{addition} and \emph{multiplication}, and satisfies the following axioms:
\begin{enumerate}[label=\textbf{(\Alph*)}]
    \item Axioms for addition
    \begin{itemize}
        \item $x + y \in F$ for all $x, y \in F$.
        \item $x + y = y + x$ for all $x, y \in F$.
        \item $(x + y) + z = x + (y + z)$ for all $x, y, z \in F$.
        \item There exists an element $0 \in F$ such that $x + 0 = x$ for all $x \in F$.
        \item For each $x \in F$, there exists an element $-x \in F$ such that $x + (-x) = 0$.
    \end{itemize}
    \item Axioms for multiplication
    \begin{itemize}
        \item $x \cdot y \in F$ for all $x, y \in F$.
        \item $x \cdot y = y \cdot x$ for all $x, y \in F$.
        \item $(x \cdot y) \cdot z = x \cdot (y \cdot z)$ for all $x, y, z \in F$.
        \item There exists an element $1 \in F$ with $1 \neq 0$ such that $x \cdot 1 = x$ for all $x \in F$.
        \item For each $x \in F$ with $x \neq 0$, there exists an element $x^{-1} \in F$ such that $x \cdot x^{-1} = 1$.
    \end{itemize}
    \item The distributive law
    \begin{itemize}
        \item $x \cdot (y + z) = x \cdot y + x \cdot z$ for all $x, y, z \in F$.
    \end{itemize}
\end{enumerate}

A set $F$ is called an \textbf{ordered field} if it is both an ordered set and a field, and satisfies the following axioms:
\begin{itemize}
    \item If $x, y, z \in F$ and $x < y$, then $x + z < y + z$.
    \item If $x, y \in F$ and $x > 0$ and $y > 0$, then $x \cdot y > 0$.
\end{itemize}

\bigskip

\begin{definition} \label{def:Boundedness}
    Let $S$ be an ordered set and $E \subseteq S$.
    \begin{itemize}
        \item $E$ is \textbf{bounded above} if there exists some $M \in S$
        such that $x \leq M$ for all $x \in E$. Such an $M$ is called an \textbf{upper bound} of $E$.
        \item $E$ is \textbf{bounded below} if there exists some $m \in S$
        such that $m \leq x$ for all $x \in E$. Such an $m$ is called a \textbf{lower bound} of $E$.
    \end{itemize}
\end{definition}

\bigskip

\begin{theorem} \label{thm:BoundedSubsetOfZinZHasExtremum}
    Any nonempty subset of $\Z$ which is bounded below in $\Z$ has a unique least element,
    and any nonempty subset of $\Z$ which is bounded above in $\Z$ has a unique greatest element.
\end{theorem}

\begin{proof}
    Let $E$ be a nonempty subset of $\Z$ which is bounded below in $\Z$.
    Then there exists a lower bound $m \in \Z$ of $E$. Let $F$ be defined as follows:
    \[
    F \coloneqq \setbuilder{x - m + 1}{x \in E} \subseteq \N.
    \]
    By \Cref{thm:WellOrderingProperty}, $F$ has a least element $n$. Then we have
    \[
    n \leq x - m + 1 \text{ for all } x \in E \implies n + m - 1 \leq x \text{ for all } x \in E.
    \]
    Since $n \in F$, there exists a unique $x_{0} \in E$ such that $n = x_{0} - m + 1$,
    so $x_{0} = n + m - 1$ is the unique least element of $E$.
    \\[6mm]
    Let $S$ be a nonempty subset of $\Z$ which is bounded above in $\Z$.
    Then there exists an upper bound $M \in \Z$ of $S$. Let $T$ be defined as follows:
    \[
    T \coloneqq \setbuilder{-x + M + 1}{x \in S} \subseteq \N.
    \]
    By \Cref{thm:WellOrderingProperty}, $T$ has a least element $n$. Then we have
    \[
    n \leq -x + M + 1 \text{ for all } x \in S \implies M - n + 1 \geq x \text{ for all } x \in S.
    \]
    Since $n \in T$, there exists a unique $x_{0} \in S$ such that $n = -x_{0} + M + 1$,
    so $x_{0} = M - n + 1$ is the unique greatest element of $S$.
\end{proof}

\bigskip

\begin{definition} \label{def:SupremumInfimum}
    Let $S$ be an ordered set. Assume that $E \subseteq S$ is bounded above. If there exists an $\alpha \in S$ such that
    \begin{itemize}
        \item $\alpha$ is an upper bound of $E$.
        \item If $\beta < \alpha$, then $\beta$ is not an upper bound of $E$.
    \end{itemize}
    then $\alpha$ is called the \textbf{supremum} (or \emph{least upper bound}) of $E$, denoted by
    \[
    \alpha = \sup E.
    \]
    Similarly, Assume that $F \subseteq S$ is bounded below. If there exists an $\gamma \in F$ such that
    \begin{itemize}
        \item $\gamma$ is a lower bound of $F$.
        \item If $\beta > \gamma$, then $\beta$ is not a lower bound of $F$.
    \end{itemize}
    then $\gamma$ is called the \textbf{infimum} (or \emph{greatest lower bound}) of $F$, denoted by
    \[
    \gamma = \inf F.
    \]
\end{definition}

\bigskip

\begin{example}
    Let $S \subseteq \Q^{+}$ be defined as
    \[
    S \coloneqq \setbuilder{p \in \Q^{+}}{p^{2} < 2}.
    \]
    For any $p \in S$, let
    \[
    q \coloneqq p - \dfrac{p^{2} - 2}{p + 2} = \dfrac{2p + 2}{p + 2} > p.
    \]
    Then we have $q \in \Q^{+}$ and
    \[
    q^{2} - 2 = \dfrac{2(p^{2} - 2)}{(p + 2)^{2}} < 0 \implies q \in S.
    \]
    This implies that any $p \in S$ is not an upper bound of $S$.
    \\[6mm]
    For any $p \in \Q^{+}$ with $p^{2} > 2$, $p$ is an upper bound of $S$. Let 
    \[
    q \coloneqq p - \dfrac{p^{2} - 2}{p + 2} = \dfrac{2p + 2}{p + 2} < p.
    \]
    Then we have $q \in \Q^{+}$ and
    \[
    q^{2} - 2 = \dfrac{2(p^{2} - 2)}{(p + 2)^{2}} > 0 \implies q \text{ is a lower bound of } S.
    \]
    This implies that such $p$ is not the least upper bound of $S$.
    Since there exists no $p \in \Q^{+}$ such that $p^{2} = 2$, $S$ has no least upper bound in $\Q^{+}$.
\end{example}

\bigskip

\begin{definition} \label{def:LeastUpperBoundProperty}
    An ordered set $S$ is said to have the \textbf{least upper bound property}
    if any nonempty subset of $S$ which is bounded above has a least upper bound in $S$.
\end{definition}

\bigskip

\begin{theorem} \label{thm:GreatestLowerBoundProperty}
    Let $S$ be an ordered set with the least upper bound property. Then, any nonempty subset of $S$
    which is bounded below has a greatest lower bound in $S$.
\end{theorem}

\begin{proof}
    Let $E$ be a nonempty subset of $S$ which is bounded below. Let $L$ be defined as follows:
    \[
    L \coloneqq \setbuilder{\lambda}{\lambda \text{ is a lower bound of } E}.
    \]
    Then, $L$ is nonempty and bounded above. By the least upper bound property,
    there exists $\alpha = \sup L \in S$. We will show that $\alpha = \inf E$.
    \\[6mm]
    Assume that $\alpha \notin L$. Then we have the following:
    \begin{itemize}
        \item $\alpha$ is not a lower bound of $E \implies$ There exists $\beta \in E$ such that $\beta < \alpha$.
        \item $\alpha = \sup L \implies \beta$ is not an upper bound of $L \implies$
        There exists $\gamma \in L$ such that $\beta < \gamma \leq \alpha$.
        \item $\gamma \in L \implies \gamma$ is a lower bound of $E \implies \gamma \leq \beta$.
    \end{itemize}
    This is a contradiction. Therefore, we have $\alpha \in L$, i.e., $\alpha$ is a lower bound of $E$.
    Since $\alpha$ is an upper bound of $L$, the following holds:
    \[
    \text{For any } \delta > \alpha, \text{ we have } \delta \notin L
    \implies \delta \text{ is not a lower bound of } E.
    \]
    Therefore, we can conclude that $\alpha = \inf E$. 
\end{proof}
